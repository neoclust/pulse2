76 def updateQueryClass():
77  """
78  Our fix for SQLAlchemy behaviour with MySQL
79  Sometimes we lost the connection to the MySQL Server, even on a local
80  server.
81  """
82  q = sqlalchemy.orm.query.Query
83  for m in ['first', 'count', 'all', '__iter__']:
84  setattr(q, '_old_'+m, getattr(q, m))
85  setattr(q, m, create_method(m))
86 
87 def create_method(m):
88  def method(self, already_in_loop = False):
89  NB_DB_CONN_TRY = 2
90  NORESULT = "__noresult__"
91  ret = NORESULT
92  try:
93  old_m = getattr(self, '_old_'+m)
94  ret = old_m()
95  except DBAPIError, e:
96  reconnect = False
97  if e.orig.args[0] == 2013 and not already_in_loop: # Lost connection to MySQL server during query error
98  logging.getLogger().warn("DBAPIError Lost connection")
99  reconnect = True
100  elif e.orig.args[0] == 2006 and not already_in_loop: # MySQL server has gone away
101  logging.getLogger().warn("DBAPIError MySQL server has gone away")
102  reconnect = True
103  if reconnect:
104  for i in range(0, NB_DB_CONN_TRY):
105  logging.getLogger().warn("Trying to recover the connection (try #%d on %d)" % (i + 1, NB_DB_CONN_TRY + 1))
106  new_m = getattr(self, m)
107  try:
108  ret = new_m(True)
109  break
110  except Exception, e:
111  # Try again
112  continue
113  if ret != NORESULT:
114  return ret
115  raise e
116  return ret
117  return method

209 def updateDebianSourceList():
210  try:
211  installation_uuid = open('/etc/pulse-licensing/installation_id').read().strip()
212  except IOError:
213  logging.getLogger().error('Error while reading installation_id file')
214  try:
215  pulse_version = getVersion().split('.')[0]
216  # Pulse repository line
217  repo_line = 'deb http://%s:a0@pulse.mandriva.org/pub/pulse2/server/debian wheezy %s.0\n' % (installation_uuid, pulse_version)
218 
219  lines = open('/etc/apt/sources.list', 'r').readlines()
220  for i in xrange(len(lines)):
221  line = lines[i]
222  # If there is already a pulse line, we overwrite it (skip comment line)
223  if 'pulse.mandriva.org/pub/pulse2/server/debian' in line and not '#' in line:
224  lines[i] = repo_line
225  break
226  else:
227  lines.append(repo_line)
228 
229  # Writing file
230  f = open('/etc/apt/sources.list', 'w')
231  f.writelines(lines)
232  f.close()
233  except IOError:
234  logging.getLogger().error('Error while writing source.list file')
235  except Exception, e:
236  logging.getLogger().exception(str(e))
