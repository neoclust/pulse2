72  fobj = file(fname, 'w+')

101  def _gotScheduler(self, result):
102  """
103  Called when a scheduler has been found to start the download
104  """
105  if not result:

112  # Create the lock file

114  f.close()
115  # Start download process
116  d = getProxy(MscConfig().schedulers[scheduler_name]).callRemote(

189  def getFilesList(self):
190  """
191  Return the available downloaded files for the current user.
192  The later the latest in the list.
193 
194  @rtype: list
195  @returns: list of tuples (filename, uuid, timestamp, length, inode)
196  """
197  ret = []
198  if os.path.exists(self.storage):
199  tmp = {}
200  # Get files list from the storage directory
201  for fname in os.listdir(self.storage):
202  statinfo = os.stat(os.path.join(self.storage, fname))
203  if fname.endswith(self.LOCKEXT):
204  # Download lock file
205  timestamp = statinfo.st_mtime
206  uuid = fname.split('.')[0]
207  tmp[timestamp] = ('', uuid, list(datetime.datetime.fromtimestamp(float(statinfo.st_mtime)).timetuple()), 0, statinfo.st_ino)
208  elif fname.endswith(self.ERROREXT):
209  # Error status file
210  uuid, timestamp = os.path.splitext(fname)[0].split('-')
211  tmp[timestamp] = ('', uuid, list(datetime.datetime.fromtimestamp(float(timestamp)).timetuple()), -1, statinfo.st_ino)
212  else:
213  # Downloaded file
214  try:
215  uuid, timestamp, name = fname.split('-', 2)
216  tmp[int(timestamp)] = (name, uuid, list(datetime.datetime.fromtimestamp(float(timestamp)).timetuple()), statinfo.st_size, statinfo.st_ino)
217  except ValueError:
218  pass
219  # Sort file list by inverse chronological order
220  timestamps = tmp.keys()
221  timestamps.sort(reverse = True)
222  for timestamp in timestamps:
223  ret.append(tmp[timestamp])
224  return ret
