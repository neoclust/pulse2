148  def setoption(self, section, key, attrib, type = 'str'):
149  if type == 'str':
150  if self.cp.has_option(section, key):
151  setattr(self, attrib, self.cp.get(section, key))
152  log.debug("launcher %s: section %s, option %s set to '%s'" % (self.name, section, key, getattr(self, attrib)))
153  else:
154  log.debug("launcher %s: section %s, option %s not set, using default value '%s'" % (self.name, section, key, getattr(self, attrib)))
155  elif type == 'bool':
156  if self.cp.has_option(section, key):
157  setattr(self, attrib, self.cp.getboolean(section, key))
158  log.debug("launcher %s: section %s, option %s set to %s" % (self.name, section, key, getattr(self, attrib)))
159  else:
160  log.debug("launcher %s: section %s, option %s not set, using default value %s" % (self.name, section, key, getattr(self, attrib)))
161  elif type == 'int':
162  if self.cp.has_option(section, key):
163  setattr(self, attrib, self.cp.getint(section, key))
164  log.debug("launcher %s: section %s, option %s set to %s" % (self.name, section, key, getattr(self, attrib)))
165  else:
166  log.debug("launcher %s: section %s, option %s not set, using default value %s" % (self.name, section, key, getattr(self, attrib)))
167  elif type == 'pass':
168  if self.cp.has_option(section, key):
169  setattr(self, attrib, self.cp.getpassword(section, key))
170  log.debug("launcher %s: section %s, option %s set using given value" % (self.name, section, key))
171  else:
172  log.debug("launcher %s: section %s, option %s not set, using default value" % (self.name, section, key))

429  def conf_or_default(self, type):
430  label = type.upper().replace("_", " ")
431  conf = getattr(self, "%s_path"%(type))
432  default = getattr(self, "%s_path_default"%(type))
