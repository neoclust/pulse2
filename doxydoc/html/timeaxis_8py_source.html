189  @classmethod
190  def extract_intervals(cls, deployment_intervals):
191  """
192  Extracts the intervals from string to the list of tuples.
193 
194  @param deployment_intervals: intervals on format i.e. "10-14,20-5"
195  @type deployment_intervals: str
196 
197  @return: normalized collection of intervals
198  @rtype: list
199  """
200  intervals = []
201  for segment in deployment_intervals.split(','):
202  unit = segment.split('-')
203  if len(unit) == 2:
204  (start, end) = [int(d) for d in unit]
205  if start > end:
206  intervals.append((start, 0))
207  intervals.append((0, end))
208  elif start == end and start != 0:
209  raise ValueError("<start> must be less or great than to <end>")
210  else:
211  intervals.append((start, end))
212  return intervals
