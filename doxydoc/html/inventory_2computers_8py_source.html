101  def __restrictLocationsOnImagingServerOrEntity(self, filt, ctx):
102  if 'imaging_server' in filt and filt['imaging_server'] != '':

106  if main_imaging_entity_uuid != None:

108  filt['imaging_entities'] = [main_imaging_entity_uuid]
109  self.logger.debug('Get now children entities of this main imaging entity')
110  # Get childs entities of this main_imaging_entity_uuid
111  # Search only in user context
112  for loc in self.inventory.getUserLocations(ctx.userid):
113  if ComputerImagingManager().isChildOfImagingServer(loc.uuid, main_imaging_entity_uuid):
114  self.logger.debug('Found %s as child entity of %s' % (loc.uuid, main_imaging_entity_uuid))
115  filt['imaging_entities'].append(loc.uuid)
116  else:
117  self.logger.warn("can't get the entity that correspond to the imaging server %s"%(filt['imaging_server']))
118  return [False]
119 
120  if 'imaging_entities' in filt:

122  for l in ctx.locations:

164  logging.getLogger().debug("Computer %s cannot be added in an imaging group:" % network['cn'])
