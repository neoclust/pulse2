230  def __allgroups_query(self, ctx, params, session = None, type = 0):
231  """
232  return a query on group
233  filtered by several flags like canShow, dynamic, static, type (0 = group, 1 = profile)
234  and filters like filter (on the group name with wildcards) and name (on the group name without wildcards)
235  """
236  (select_from, filter_on) = self.__get_group_permissions_request_first(ctx, session)
237  groups = session.query(Groups).add_column(self.users.c.login).select_from(select_from).filter(self.groups.c.type == type)
238  root_user = self.__getUser('root')
239  if filter_on is not None:
240  groups = groups.filter(filter_on)
241 
242  if ctx.userid == 'root' \
243  and 'localSidebar' in params \
244  and params['localSidebar'] \
245  and root_user:
246  groups = groups.filter(self.groups.c.FK_users == root_user.id)
247 
248  if 'canShow' in params:
249  if params['canShow']:
250  groups = groups.filter(self.shareGroup.c.display_in_menu == 1)
251  else:
252  groups = groups.filter(self.shareGroup.c.display_in_menu == 0)
253 
254  try:
255  if params['owner']:
256  groups = groups.filter(self.users.c.login == params['owner'])
257  except KeyError:
258  pass
259 
260  try:
261  if params['dynamic']:
262  groups = groups.filter(self.groups.c.query != None)
263  except KeyError:
264  pass
265 
266  try:
267  if params['static']:
268  groups = groups.filter(self.groups.c.query == None)
269  except KeyError:
270  pass
271 
272  try:
273  if params['filter']:
274  groups = groups.filter(self.groups.c.name.like('%'+params['filter'].encode('utf-8')+'%'))
275  except KeyError:
276  pass
277 
278  try:
279  if params['name'] != None:
280  groups = groups.filter(self.groups.c.name == params['name'].encode('utf-8'))
281  except KeyError:
282  pass
283 
284  return groups.group_by(self.groups.c.id)

298  def getallgroups(self, ctx, params, type = 0):
299  """
300  get all group filtered by the params dict
301  flags can be canShow, dynamic, static, type (0 = group, 1 = profile)
302  and filters can be filter (on the group name with wildcards) and name (on the group name without wildcards)
303  it's also possible to use min and max to only have a page of the results
304  """
305  session = create_session()
306  groups = self.__allgroups_query(ctx, params, session, type)
307  groups = groups.order_by(self.groups.c.name)
308 
309  min = 0
310  try:
311  if params['min']:
312  min = int(params['min'])
313  groups = groups.offset(int(min))
314  except KeyError:
315  pass
316 
317  try:
318  if params['max'] != -1:
319  max = int(params['max']) - min
320  groups = groups.limit(max)
321  except KeyError:
322  pass
323 
324  result = groups.all()
325  ret = []
326  # For each group check if the user is the owner of the group
327  # (ie; the user that has created the group)
328  for m in result:
329  if m[1] == ctx.userid:
330  m[0].is_owner = True
331  ret.append(m[0])
332 
333  session.close()
334  return ret

418  @DatabaseHelper._session
419  def delete_group(self, session, ctx, id):
420  """
421  delete a group defined by it's id
422  delete the results and the machines linked to that group if needed
423  """
424  # Is current group a profile ?
425  if self.isprofile(ctx, id):
426  resultTable = self.profilesResults
427  resultKlass = ProfilesResults
428  else:
429  resultTable = self.results
430  resultKlass = Results
431 
432  self.__getOrCreateUser(ctx)
433  connection = self.getDbConnection()
434  trans = connection.begin()
435  # get machines to possibly delete
436  to_delete = [x.id for x in session.query(Machines).select_from(self.machines.join(resultTable)).filter(resultTable.c.FK_groups == id)]
437  # Delete the previous results for this group in the Results table
438  session.query(resultKlass).filter_by(FK_groups = id).delete()
439  # Delete all shares on the group before delete group
440  self.__deleteShares(id, session)
441  # Update the Machines table to remove ghost records
442  self.__updateMachinesTable(connection, to_delete)
443  # Delete the group from the Groups table
444  session.query(ProfilesData).filter_by(FK_groups = id).delete()
445  session.query(Groups).filter_by(id = id).delete()
446  # Delete convergence groups for this id
447  self.delete_convergence_groups(id)
448  session.flush()
449  trans.commit()
450  return True

647  def __permissions_query(self, ctx, session):
648  """
649  get all the parts to build a query to get only the group which the user can have access
650  """
651  join_tables = [[self.users, False, self.users.c.id==self.groups.c.FK_users],
652  [self.shareGroup, True, self.groups.c.id==self.shareGroup.c.FK_groups]]
653  filters = None
654 
655  if ctx.userid<>'root':
656  # Filter on user groups
657  user_id = self.__getOrCreateUser(ctx)
658  filters = [ self.users.c.login==ctx.userid,
659  self.shareGroup.c.FK_users==user_id ]
660 
661  # get all usergroups ids
662  ug_ids = []
663  for user in self.__getUsers(getUserGroups(ctx.userid), 1, session):
