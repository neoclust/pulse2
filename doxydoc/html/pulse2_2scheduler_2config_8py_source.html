133  def setoption(self, section, key, attrib, type = 'str'):
134  if type == 'str':
135  if self.cp.has_option(section, key):
136  setattr(self, attrib, self.cp.get(section, key))

138  else:
139  log.debug("scheduler %s: section %s, option %s not set, using default value '%s'" % (self.name, section, key, getattr(self, attrib)))
140  elif type == 'bool':

145  log.debug("scheduler %s: section %s, option %s not set, using default value %s" % (self.name, section, key, getattr(self, attrib)))
