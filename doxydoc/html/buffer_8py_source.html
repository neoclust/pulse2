42 def initialized(method):
43  """
44  Blocks the decorated method when _initialized attribute is False.

46  @param method: decorated method

98  @initialized
99  def backup_buffer(self):
100  """
101  When the scheduler-proxy service ends, internal buffer may contain
102  some responses to send. This runtime ensures a backup to temp file,
103  which can be restored when sheduler-proxy starts.
104  """
105  if len(self.packets) > 0 :
106  self.logger.info("XMLRPC Proxy: Backup the buffer with %d responses"
107  % len(self.packets))
108  with open(self.config.scheduler_proxy_buffer_tmp, "wb") as fp :
109  pickle.dump(self.packets, fp)
110 
111  @initialized
112  def restore_buffer(self):
113  """
114  Restoring the backuped responses from a temp file.
115 
116  When the scheduler-proxy service starts, this runtime checks
117  the content of temp file for unsent responses.
118  This content is moved to internal buffer to resend.
119  """
120  try:
121  if os.path.exists(self.config.scheduler_proxy_buffer_tmp):
122  self.logger.info("XMLRPC Proxy: Restoring the buffer")
123 
124  with open(self.config.scheduler_proxy_buffer_tmp, "rb") as fp :
125  try:
126  content = pickle.load(fp)
127  except Exception, e:
128  self.logger.warn("XMLRPC Proxy: An error occured when restoring the buffer: %s" % str(e))
129  if isinstance(content, list):
130  self.packets.extend(content)
131  else:
132  self.logger.warn("XMLRPC Proxy: Invalid format of backup of buffer, operation ignored")
133 
134  os.unlink(self.config.scheduler_proxy_buffer_tmp)
135  self.logger.info("XMLRPC Proxy: restore buffer with %d responses" % len(SendingBuffer().packets))
136  except Exception, exc:
137  self.logger.error("XMLRPC Proxy: buffer restore failed: %s" % str(exc))
