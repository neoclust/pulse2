98 class HTTPClientFactoryWithHeader(HTTPClientFactory):
99  """
100  HTTPClientFactory don't allow to get the HTTP header.
101  So we subclass the page() method and modify it to get the HTTP payload
102  header
103  """
104 
105  # the GLPI anti-csrf token (GLPI 0.83.3+)
106  glpi_csrf_token = ''
107 
108  def page(self, page):
109  # grabbing the GLPI anti-csrf token (GLPI 0.83.3+) by
110  # looking for such patterns :
111  # <input type='hidden' name='_glpi_csrf_token' value='82d37af7f30d76f2238d49c28167654f'>
112  m = re.search("<input type='hidden' name='_glpi_csrf_token' value='([0-9a-z]{32})'>", page)
113  if m is not None:
114  self.glpi_csrf_token = m.group(1)
115  if self.waiting:
116  self.waiting = 0
117  self.deferred.callback(self)

120 def getPageWithHeader(url, contextFactory=None, *args, **kwargs):
121  """
122  Same as twisted.web.client.getPage, but we keep the HTTP header in the
123  result thanks to the HTTPClientFactoryWithHeader class
124  """
125  if parseAvailable:
126  scheme, host, port, path = _parse(url)
127  else:
128  uri = _URI.fromBytes(url)
129  scheme = uri.scheme
130  host = uri.host
131  port = uri.port
132 
133  factory = HTTPClientFactoryWithHeader(url, *args, **kwargs)
134  d = factory.deferred
135 
136  if scheme == 'https':
137  from twisted.internet import ssl
138  if contextFactory is None:
139  contextFactory = ssl.ClientContextFactory()
140  reactor.connectSSL(host, port, factory, contextFactory)
141  else:
142  reactor.connectTCP(host, port, factory)
143  return d
