674  def __queryUsersFilterBis(self, ctx):
675  """
676  Build a part of a query for commands, that add user filtering
677  """
678  if ctx.filterType == "mine":
679  # User just want to get her/his commands
680  return self.commands.c.creator == ctx.userid
681  elif ctx.filterType == "all":
682  # User want to get all commands she/he has the right to see
683  if ctx.userid == "root":
684  # root can see everything, so no filter for root
685  return True
686  elif ctx.locationsCount not in [None, 0, 1] and ctx.userids:
687  # We have multiple locations, and a list of userids sharing the
688  # same locations of the current user
689  userids = ctx.userids
690  # If show root commands is activated, we add it
691  if self.config.show_root_commands:
692  userids.append('root')
693  return self.commands.c.creator.in_(userids)
694  else:
695  # Unknown filter type
696  self.logger.debug("Unknown filter type when querying commands")
697  if ctx.locationsCount not in [None, 0, 1]:
698  # We have multiple locations (entities) in database, so we
699  # filter the results using the current userid
700  return self.commands.c.creator == ctx.userid
701  return True
