24 """
25  get a command list
26  give a CSV-ouput of the command advancement on its targets, using the history table
27  lines are formated like this:
28  - timestamp (rounded 60 seconds)
29  - amount of upload/exec/delete in progress during those 60 seconds
30  - amount of upload/exec/delete failed during those 60 seconds
31  - amount of upload/exec/delete done during those 60 seconds
32  - ammount of targets in stage 1 (upload done, exec todo)
33  - ammount of targets in stage 2 (exec done, delete todo)
34  - ammount of targets in stage 3 (delete done)
35 """

40 from sqlalchemy import create_engine, MetaData, Table, Column, Integer, ForeignKey, select
41 
42 if len(sys.argv) != 3:
43  print "usage: %s <mysql-uri> <command-id-comma-separated>" % sys.argv[0]
44  exit(1)

46 # create connection
47 mysql_db = create_engine(sys.argv[1])
48 metadata = MetaData(mysql_db)
49 
50 coh_table = Table('commands_on_host',
51  metadata,
52  autoload = True)
53 history_table = Table("commands_history",
54  metadata,
55  Column('fk_commands_on_host', Integer, ForeignKey('commands_on_host.id')),
56  autoload = True)
57 
58 # prepare command ids
59 ids = sys.argv[2].split(',')
60 
61 hist_data = select([
62  history_table.c.fk_commands_on_host,
63  history_table.c.date,
64  history_table.c.state,
65  history_table.c.error_code
66  ]
67  ).select_from(
68  history_table.join(coh_table)
69  ).where(
70  coh_table.c.fk_commands.in_(ids)
71  ).order_by(history_table.c.date).execute().fetchall()
