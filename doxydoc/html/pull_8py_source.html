48  def parse_pull_phase_result(self,(exitcode, stdout, stderr)):
49 
50  if exitcode == PULSE2_SUCCESS_ERROR: # success
51  self.logger.info("Circuit #%s: pull %s done (exitcode == 0)" % (self.coh.id, self.name))
52  self.update_history_done(exitcode, stdout, stderr)
53  if self.coh.isStateStopped():
54  return DIRECTIVE.KILLED
55 
56  if self.phase.switch_to_done():
57  return self.next()
58  return self.give_up()
59 
60  elif self.name in self.config.non_fatal_steps:
61  self.logger.info("Circuit #%s: pull %s failed (exitcode != 0), but non fatal according to scheduler config file" % (self.coh.id, self.name))
62  self.update_history_failed(exitcode, stdout, stderr)
63  self.phase.set_done()
64  return self.next()
65 
66  else: # failure: immediately give up
67  self.logger.info("Circuit #%s: pull %s failed (exitcode != 0)" % (self.coh.id, self.name))
68  self.update_history_failed(exitcode, stdout, stderr)
69  return self.switch_phase_failed()
