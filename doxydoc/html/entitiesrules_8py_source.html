45  def _readRulesFile(self):
46  """
47  Read the rules configuration file
48  """
49  self.logger.debug("Reading inventory rules file %s" % self.conf)
50  for line in file(self.conf):
51  if line.startswith('#') or not line.strip():
52  continue
53  try:
54  # The first column may contain the quoted entity list
55  m = re.search('^"(.+)"\W+(.*)$', line)
56  if m:
57  entities = m.group(1)
58  rule = m.group(2)
59  else:
60  entities, rule = line.split(None, 1)
61  entitieslist = entities.split(',')
62  for entity in entitieslist:
63  checkEntityName(entity)
64  if entitieslist:
65  words = rule.split()
66  prefix = 'none'
67  subexprs = []
68  while words:
69  if words[0] in ['and', 'or']:
70  if prefix == 'none':
71  prefix = words[0].lower()
72  words = words[1:]
73  else:
74  raise Exception('Different operators are not supported for a rule')
75  else:
76  if len(words) < 3:
77  raise Exception('Malformed rule')
78  else:
79  operand1, operator, operand2 = words[0:3]
80  operator = operator.lower()
81 
82  if operator in self.operators:
83  # TODO: Maybe check operand1 value
84  if operator == 'match':
85  # Try to compile the regexp
86  regexp = re.compile(operand2)
87  subexprs.append((operand1, operator, regexp))
88  else:
89  subexprs.append((operand1, operator, operand2))
90  else:
91  self.logger.error("Operator %s is not supported, skipping" % operator)
92  words = words[3:]
93  self.rules.append((entitieslist, prefix, subexprs))
94  except Exception:
95  self.logger.error("Error while reading this rule: %s" % line)
96  raise

106  """
107  Return the values of the given parameter from input, or an empty list
108  if no value found.
109  In this implementation, input must be the inventory of a computer (i.e.

111  """

121  def compute(self, input):
122  """
123  Returns an entity list according to the given input and the current
124  rules.
125  """
126  self.reload_file_rule()
127  ret = []
128  for entities, mainop, rules in self.rules:
129  result = None
130  for rule in rules:
131  operand1, operator, operand2 = rule
132  # Get the values of the first operand
133  values = self._getValues(input, operand1)
134  #if operand1 network/ip value tab des ips
135  if values == []:
136  # No corresponding value found, we break the loop
137  self.logger.debug("No corresponding value found for operand '%s', skipping the line" % operand1)
138  break
139  # Loop over all the values, and break the loop if one value
140  # makes the expression returns True
141  tmpresult = False
142  for value in values:
143  if operator == 'match':
144  tmpresult = operand2.match(value) != None
145  elif operator == 'equal':
146  tmpresult = value == operand2
147  elif operator == 'noequal':
148  tmpresult = value != operand2
149  elif operator == 'contains':
150  tmpresult = operand2 in value
151  elif operator == 'nocontains':
152  tmpresult = not (operand2 in value)
153  elif operator == 'starts':
154  tmpresult = value.startswith( operand2 )
155  elif operator == 'finishes':
156  tmpresult = value.endswith( operand2 )
157  else:
158  pass
159  if tmpresult:
160  self.logger.info('operator [%s] %s %s %s' %(operator, values, operand2, tmpresult))
161  break
162 
163  if mainop == 'none':
164  result = tmpresult
165  elif mainop == 'and':
166  if not tmpresult:
167  result = False
168  # Exit AND because we have a False
169  break
170  else:
171  result = True
172  elif mainop == 'or':
173  if result != None:
174  result = result or tmpresult
175  else:
176  result = tmpresult
177  else:
178  pass
179  if result:
180  ret = entities
181  # We exit as soon as a line match
182  break
183  return ret
