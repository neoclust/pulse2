33 def setupSSHAgent():
34  ssh_agent = popen2.Popen4("ssh-agent") # FIXME: should be render configurable
35  ssh_agent_auth_sock_re = "^SSH_AUTH_SOCK=([^;]+);";
36  ssh_agent_pid_re = "^SSH_AGENT_PID=([^;]+);";
37 
38  if not ssh_agent.wait() == 0: # exit code != 0: tell it
39  logging.getLogger().warn("launcher %s: Couldn't run the ssh-agent binary, hope you won't need to forward ssh keys !" % (LauncherConfig().name))
40  return
41 
42  for line in ssh_agent.fromchild.readlines():
43  if re.match(ssh_agent_auth_sock_re, line):
44  LauncherConfig().ssh_agent_sock = re.search(ssh_agent_auth_sock_re, line).group(1)
45  if re.match(ssh_agent_pid_re, line):
46  LauncherConfig().ssh_agent_pid = int(re.search(ssh_agent_pid_re, line).group(1))
47 
48  if not LauncherConfig().ssh_agent_sock:
49  logging.getLogger().info("launcher %s: Successfully run the ssh-agent binary but can't find auth socket" % (LauncherConfig().name))
50  return
51 
52  os.putenv('SSH_AUTH_SOCK', LauncherConfig().ssh_agent_sock)
53  logging.getLogger().info("launcher %s: Successfully run the ssh-agent binary, auth socket is %s" % (LauncherConfig().name, LauncherConfig().ssh_agent_sock))
54 
55  for keyname in LauncherConfig().ssh_keys:
56  if addPrivKeyToSSHAgent(keyname):
57  logging.getLogger().info("launcher %s: Successfuly declared the ssh key '%s' to your ssh agent" % (LauncherConfig().name, keyname))
58  else:
59  logging.getLogger().warn("launcher %s: Couldn't declare the ssh key '%s' to your ssh agent, hope you won't need to forward it !" % (LauncherConfig().name, keyname))

61 def killSSHAgent():
62  if LauncherConfig().ssh_agent_pid:
63  logging.getLogger().info("launcher %s: terminating the ssh-agent binary (pid is %s)" % (LauncherConfig().name, LauncherConfig().ssh_agent_pid))
64  try:
65  os.kill(LauncherConfig().ssh_agent_pid, signal.SIGTERM)
66  time.sleep(1) # give ssh-agent 1 second to exit
67  try:
68  os.kill(LauncherConfig().ssh_agent_pid, signal.SIGKILL)
69  logging.getLogger().warn("launcher %s: Had to kill the ssh-agent binary" % (LauncherConfig().name))
70  except OSError: # pid do not exists anymore
71  logging.getLogger().info("launcher %s: Successfully terminated the ssh-agent binary" % (LauncherConfig().name))
72  except OSError: # agent do not exists anymore
73  logging.getLogger().warn("launcher %s: the ssh-agent binary seems already down" % (LauncherConfig().name))

75 def addPrivKeyToSSHAgent(key_name):
76  """
77  ask the ssh-agent to keep our key
78 
79  """
80  if key_name == None or key_name == '':
81  key_name = LauncherConfig().ssh_defaultkey
82 
83  if key_name not in LauncherConfig().ssh_keys.keys():
84  return False
85
