30 class ParabolicBalance (object):
31  """
32  A interpretation of run of the parabolic curve, based
33  on quadratic function : y = - ax^2 + bx.
34 
35  The balance coeficient is based on matematical integration
36  of area between the parabolic curve and x-axis.
37 
38  Output is a list of balance coeficients which total is 1.
39  """
40  delta = 1
41 
42  # Example for 5 attepts :
43  # -----------------------
44  #
45  # |
46  # | .--^^^--. -x^2 + kx
47  # | :° | | °:
48  # | ° | | °
49  # | / | | \
50  # | .° | | °.
51  # | / | | | | \
52  # | . | | | | .
53  # |/ S0 | S1 | S2 | S3 | S4 \
54  # +-----+-----+-----+-----+-----+---
55  # 1 2 3 4 k=5
56  # ---- number of attempts ----->
57  #
58  # S0..S4 - balance coefficients

69  def get_index (self, n):
70  """ """
71  if n in range(self.attempts_total+1) :
72  return self.delta * n
73  else :
74  logging.getLogger().debug("ParabolicBalance: Out of area")

76  def fx (self, x):
77  """
78  Integral of quadratic function.
79 
80  ITG (-x^2 + kx) dx = - x^3/3 + 2kx/2
81  """
82  k = self.attempts_total
83  return ( (k * x**2 / 2.0) - x**3 / 3.0)

85  def fx_delta (self, x):
86  """
87  Get slice of integrated area.
88  """
89  return self.fx(x) - self.fx(x - self.delta)

91  def _calc(self):
92 
93  areas = []
94  for period in range(self.attempts_total) :
95  x = self.get_index(period+1)
96  s = self.fx_delta(x)
97  areas.append(s)
98 
99  total_area = sum(areas)
100  for area in areas :
101  balance = 1.0 * area / total_area
102  self._balances.append(balance)

110 def randomListByBalance (balances, limit):
111  """
112  Function to selecting the commands_on_host ids to re-schedule.
113 
114  First step is the choice of treshold into the sorted dict (by balance).
115  This treshold is considered as the lower limit of the interval to drawing
116  next ids to select.
117  All the choices are based on random selecting.
118 
119  @param balances: dictionary of balances having CoH id as key
120  @type balances: dict
121 
122  @param limit: maximum of selected commands
123  @type limit: int
124 
125  @return: list of CoH ids to reschedule
126  @rtype: list
127 
128  """
129  if len(balances) > 0 :
130  if len(balances) <= limit :
131  # no necesity to choice the most important CoHs
132  return balances.keys()
133  while True :
134 
135  # sort by balance
136  sorted_keys = sorted(balances, key=balances.get, reverse=True)
137  # draw the treshold
138  drw_coh = random.choice(sorted_keys)
139 
140  logging.getLogger().debug("Keys to draw: %s " % str(sorted_keys))
141  treshold_idx = sorted_keys.index(drw_coh)
142  # remove all the keys bellow the treshold
143  sorted_keys = sorted_keys[-treshold_idx:]
144  logging.getLogger().debug("Reduced interval: %s " % str(sorted_keys))
145 
146  count = 0
147  selected = []
148  while True :
149  if count >= limit :
150  break
151  if count >= len(sorted_keys) :
152  break
153  drw_coh = random.choice(sorted_keys)
154  if drw_coh not in selected :
155  selected.append(drw_coh)
156  count += 1
157  if len(selected) / limit > 0.8 :
158  # OK when number of selected is more than 80% of limit
159  return selected
160  else :
161  logging.getLogger().debug("List of drawed CoHs too small. Repeat the drawing")

163 def getBalanceByAttempts (start_date, end_date, attempts_failed) :
164  """
165  Calculate the command priority.
166 
167  @param start_date: date of start of launched action
168  @type start_date: datetime
169 
170  @param end_date: date of end of launched action
171  @type end_date: datetime
172 
173  @return: balance coefficient
174  @rtype: float
175  """
176  # random values
177  P1 = 1.6
178  P2 = 2.0
179  now = time.time()
180 
181  start_timestamp = time.mktime(start_date.timetuple())
182  end_timestamp = time.mktime(end_date.timetuple())
183 
184 
185  # half-cycle timestamp
186  l = (end_timestamp + start_timestamp) / 2.0
187  # half-cycle duration
188  d = (end_timestamp - start_timestamp) / 2.0
189 
190  td = ((now - l) / d) ** P2
191  dd = P1 ** (attempts_failed * 1.0)
192 
193  return td / dd
