33 # here are the regex
34 MACADDR_RE = re.compile("^MAC Address:(.+)$") # MAC Address
35 IPADDR_RE = re.compile("^IP Address:(.+):([0-9]+)$") # IP Address:port
36 MEM_INFO_RE = re.compile("^M:([0-9a-f]+),U:([0-9a-f]+)$") # lower / upper mem
37 BUS_INFO_RE = re.compile("^B:([0-9A-Fa-f]+),f:([0-9A-Fa-f]+),v:([0-9A-Fa-f]+),d:([0-9A-Fa-f]+),c:([0-9A-Fa-f]+),s:([0-9A-Fa-f]+)$") # bus, dev, vendor, device, class, subclass
38 DISKINFO_RE = re.compile("^D:\(hd([0-9]+)\):CHS\(([0-9]+),([0-9]+),([0-9]+)\)=([0-9]+)$") # number, CHS, size
39 PARTINFO_RE = re.compile("^P:([0-9]+),t:([0-9a-f]+),s:([0-9]+),l:([0-9]+)$") # number, type, start, len
40 BIOSINFO_RE = re.compile("^S0:([^\|]*)\|([^\|]*)\|([^\|]*)$") # 3 components : vendor, version, date
41 SYSINFO_RE = re.compile("^S1:([^\|]*)\|([^\|]*)\|([^\|]*)\|([^\|]*)\|([0-9A-F]{32})$") # 5 components : vendor, desc, ??, ??, UUID (16 hex chars)
42 ENCLOS_RE = re.compile("^S3:([^\|]*)\|([0-9]+)$") # vendor, type
43 MEMSLOT_RE = re.compile("^SM:([0-9]+):([^:]*):([^:]*):([0-9]+):([0-9]+)$") # Size in MB, Form factor, Location, Type, Speed in MHZ
44 NUMCPU_RE = re.compile("^S4:([0-9]+)$") # CPU number
45 FEATCPU_RE = re.compile("^C:(.*)$") # CPU features, comma-separated
46 FREQCPU_RE = re.compile("^F:([0-9]+)$") # CPU frequency
47 FAMCPU_RE = re.compile("^cpuf [0-9]+:(.+)h$") # CPU Family
48 NETMASK_RE = re.compile("^mask:(.+)$") # Netmask
49 GATEWAY_RE = re.compile("^gateway:(.+)$") # Netmask

384 class BootInventory:
385  """
386  Class holding one inventory.
387 
388  Two functions of interest :
389  - load() => fill this object using data from the client
390  - dump() => dump a struct representing the object
391 
392  """
393  #######################
394  # this class property #
395  #######################
396  # memory info (lower/upper), in bytes
397  mem_info = {'lower' : 0, 'upper': 0}
398  # periphericals
399  bus_info = {}
400  # disks (and parts)
401  disk_info = {}
402  disk_info_i = {} # Disk info with int indexes to avoid missordering
403  # bios basic infos
404  bios_info = {'vendor': '', 'version': '', 'date' : ''}
405  # system basic infos
406  sys_info = {'manufacturer': '', 'product': '', 'version' : '', 'serial' : '', 'uuid' : ''}
407  # enclosure basic infos
408  enclos_info = {'vendor': '', 'type': ''}
409  # memory info
410  memory_info = []
411  # number of (logical) CPU, f.e 4 on a core 2 Duo
412  numcpu_info = 0
413  # CPu features, to be interpreted (that's a infamous 26 bytes array)
414  featcpu_info = []
415  # the CP frequencies
416  freqcpu_info = 0
417  # CPU Family, Initialize with "Unknown processor" => code 02
418  famcpu_code = "02"
419  # the client MAC adress
420  macaddr_info = ''
421  # the client netmask
422  netmask_info = ''
423  # the client gateway
424  gateway_info = ''
425  # the client subnet
426  subnet_info = ''
427  # the inventory IP source (not necesary the client IP)
428  ipaddr_info = {'ip': '', 'port': 0}
429  unprocessed = []
430 
431  def __init__(self, data = None):
432  """
433  Object creator
434 
435  @param data : the initial inventory data
436  """
437  if data != None :
438  self.load(data)
439 
440  def __str__(self):
441  """
442  Return a string version of this object
443  """
444  return self.dump().__str__()
445 
446  def load(self, data):
447  """
448  Load the inventory data nito the object
449 
450  @param data : the inventory data
451  """
452  current_disk = None # track the disk we are reading
453  self.memory_info = []
454 
455  for line in data: # process line per line
456  line=line.replace('\\~','')
457  line=line.replace('\~','')
458  line=line.replace('\x06','')
459  if not len(line):
460  continue
461 
462  # then attempts to process line per line
463 
464  mo = re.match(MEM_INFO_RE, line)
465  if mo :
466  self.mem_info['lower'] = int(mo.group(1), 16)
467  self.mem_info['upper'] = int(mo.group(2), 16)
468  continue
469 
470  mo = re.match(BUS_INFO_RE, line)
471  if mo :
472  bus = str(int(mo.group(1), 16))
473  dev = str(int(mo.group(2), 16))
474  vendor = int(mo.group(3), 16)
475  device = int(mo.group(4), 16)
476  cl = int(mo.group(5), 16)
477  subcl = int(mo.group(6), 16)
478  if not bus in self.bus_info:
479  self.bus_info[bus] = dict()
480  self.bus_info[bus][dev] = {
481  "vendor": vendor,
482  "device" : device,
483  "class" : cl,
484  "subclass" : subcl}
485  continue
486 
487  mo = re.match(DISKINFO_RE, line)
488  if mo :
489  num = str(int(mo.group(1), 10))
490  c = int(mo.group(2), 10)
491  h = int(mo.group(3), 10)
492  s = int(mo.group(4), 10)
493  sz = int(mo.group(4), 10)
494  lba_size = int(mo.group(5), 10)
495  self.disk_info_i[int(num)] = {
496  "C": c,
497  "H" : h,
498  "S" : s,
499  "size" : sz,
500  "lba_size" : lba_size,
501  "lba_size_mb" : lba_size*512/1000/1000,
502  "parts" : dict()}
503  self.disk_info[num] = {
504  "C": c,
505  "H" : h,
506  "S" : s,
507  "size" : sz,
508  "lba_size" : lba_size,
509  "lba_size_mb" : lba_size*512/1000/1000,
510  "parts" : dict()}
511  current_disk = num
512  continue
513 
514  mo = re.match(PARTINFO_RE, line)
515  if mo :
516  num = str(int(mo.group(1), 10))
517  t = int(mo.group(2), 16)
518  s = int(mo.group(3), 10)
519  l = int(mo.group(4), 10)
520  self.disk_info[current_disk]['parts'][num] = {
521  'type' : t,
522  'type_hex' : '%.2x' % t,
523  'start' : s,
524  'length_mb' : l*512/1000/1000,
525  'length' : l}
526  self.disk_info_i[int(current_disk)]['parts'][int(num)] = {
527  'type' : t,
528  'type_hex' : '%.2x' % t,
529  'start' : s,
530  'length_mb' : l*512/1000/1000,
531  'length' : l}
532  continue
533 
534  mo = re.match(BIOSINFO_RE, line)
535  if mo :
536  self.bios_info['vendor'] = mo.group(1)
537  self.bios_info['version'] = mo.group(2)
538  self.bios_info['date'] = mo.group(3)
539  continue
540 
541  mo = re.match(SYSINFO_RE, line)
542  if mo :
543  self.sys_info['manufacturer'] = mo.group(1)
544  self.sys_info['product'] = mo.group(2)
545  self.sys_info['version'] = mo.group(3)
546  self.sys_info['serial'] = mo.group(4)
547  self.sys_info['serial'] = re.sub(r'[^a-zA-Z0-9-]',r'',self.sys_info['serial'])
548  self.sys_info['uuid'] = mo.group(5)
549  continue
550 
551  mo = re.match(ENCLOS_RE, line)
552  if mo :
553  self.enclos_info['vendor'] = mo.group(1)
554  self.enclos_info['type'] = mo.group(2)
555  continue
556 
557  mo = re.match(MEMSLOT_RE, line)
558  if mo :
559  memslot_info = {}
560  memslot_info['size'] = int(mo.group(1), 10)
561  memslot_info['ff'] = mo.group(2)
562  memslot_info['location'] = mo.group(3)
563  memslot_info['type'] = int(mo.group(4),10)
564  memslot_info['speed'] = int(mo.group(5), 10)
565  self.memory_info += [memslot_info]
566  continue
