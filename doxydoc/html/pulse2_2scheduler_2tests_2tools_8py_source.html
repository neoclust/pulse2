36 
37  select COLUMN_NAME from information_schema.columns

147  def __new__(cls, name, bases, attrs):
148  # all table names to generate
149  tables = [t.replace('_cols','') for t in cls.__dict__.keys() if t.endswith ('_cols')]
150 
151  for attr_name in tables:
152  # gets the ORM class
153  klass = getattr(cls, "%s_class" % attr_name)
154  # columns parsing
155  cols_attr = getattr(cls, "%s_cols" % attr_name)
156  cols = cols_attr.replace("|","").replace(" ","").split("\n")
157  tab_attrs = dict([(a, None) for a in cols])
158  # columns are created and attached on ORM class
159  table_class = type(attr_name, (klass,), tab_attrs)
160  instance = table_class()
161  attrs.update({attr_name: instance,})
162 
163  return type.__new__(cls,name, bases, attrs)
