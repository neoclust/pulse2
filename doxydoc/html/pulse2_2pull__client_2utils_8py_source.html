27 class Singleton(object):
28  """
29  A thread safe Singleton implementation
30

32  pass

34  a = A.instance()

116 def normalize(addr):
117  """
118  Normalize a MAC address to UNIX style
119  """
120  # Determine which delimiter style out input is using
121  if "." in addr:
122  delimiter = "."
123  elif ":" in addr:
124  delimiter = ":"
125  elif "-" in addr:
126  delimiter = "-"
127  # Eliminate the delimiter
128  m = addr.replace(delimiter, "")
129  m = m.lower()
130  # Normalize
131  n = ":".join(["%s%s" % (m[i], m[i + 1]) for i in range(0, 12, 2)])
132  return n
