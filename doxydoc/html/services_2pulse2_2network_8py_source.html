60  @classmethod
61  def on_same_network(cls, ip, network, netmask):
62  """
63  Test if IP address can be found in network.
64 
65  @param ip: tested IP adddress
66  @type ip: string
67 
68  @param network: address of network
69  @type network: string
70 
71  @param netmask: netmask of network
72  @type netmask: string
73 
74  @return: True if IP address are in the same network.
75  @rtype: bool
76  """
77  if not cls.check_netmask(network, netmask):
78  log.warn("network %s is not matching to netmask %s !" % (network, netmask))

110  @classmethod
111  def netmask_to_cidr(cls, netmask):
112  """
113  Converting decimal dotted subnet mask to CIDR format
114 
115  i.e.: 255.255.255.0 -> 24
116 
117  @param netmask: netmask on dotted decimal format
118  @type netmask: str
119 
120  @return: number of bits (CIDR)
121  @rtype: int
122  """
123  dec_bytes = [int(n) for n in netmask.split(".")]
124 
125  count = 0
126  for byte in dec_bytes :
127  while byte != 0:
128  if byte % 2 == 1:
129  count += 1
130  byte = byte / 2
131 
132  return count

256  @classmethod
257  def get_subnets(cls, netmask):
258  """
259  For each bit returns number of subnets and number of elements.
260 
261  @param netmask: netmask to extract
262  @type netmask: str
263 
264  @return: number of subnets and last bit value for each bit
265  @rtype: list
266  """
267 
268  TOTAL_BITS = 8
269 
270  for element in NetUtils.ipv4_to_dec(netmask) :
271 
272  n_bits = SUBNET_BITS.index(element)
273  m_bits = TOTAL_BITS - n_bits
274 
275  nbr_subnets = 2 ** m_bits
276  last_bit_value = 2 ** n_bits
277 
278  yield (nbr_subnets, last_bit_value)

281  @classmethod
282  def get_valid_range(cls, nbr, last_bit, value):
283  """
284  Get the valid range according to checked value
285 
286  @param nbr: number of subnets
287  @type nbr: int
288 
289  @param last_bit: value of last bit of subnet
290  @type last_bit: int
291 
292  @param value: checked value
293  @type value: int
294 
295  @return: valid range to get network/broadcast addresses
296  @rtype: tuple
297  """
298 
299  if nbr == 256 :
300  return (0, 255)
301 
302  start = 0
303  end = 0
304 
305  for attempt in range(0, nbr) :
306 
307  end = attempt * last_bit - 1
308 
309  if value in range(start, end) :
310  return (start, end)
311  start = attempt * last_bit
312 
313  return (value, value)

316  def get_ranges(self):
317  """
318  Apply the netmask rules on checked IP address.
319  @return: range of all possibles IP adresses on format:
320  (min, max), (min, max), (min, max), (min, max)
321  @rtype: generator
322  """
323 
324  ip_dec = NetUtils.ipv4_to_dec(self.ip)
325 
326  i = 0
327  for nbr, last_bit in self.get_subnets(self.netmask) :
328 
329  value = ip_dec[i]
330 
331  octet_range = self.get_valid_range(nbr, last_bit, value)
332  yield octet_range
333 
334  i += 1

337  def network(self):
338  """
339  @return: calculated IPv4 network address
340  @rtype: str
341  """

348  @return: calculated IPv4 broadcast address

627 class IPResolversContainer :
628  """
629  Registering of all resolvers to get a correct network interface
630  of a client machine.
631  """
632  resolvers = []
633  networks = []

635  @classmethod
636  def is_resolver(cls, candidate):
637  """
638  Test if candidate is a subclass of abstract frame 'ResolvingCallable'.
639 
640  @param candidate: candidate to check
641  @type candidate: object
642 
643  @return: True if candidate is a resolver
644  @rtype: bool
645  """
646  return inspect.isclass(candidate) and issubclass(candidate, ResolvingCallable)
647 
648  @classmethod
649  def get_all_resolvers(cls):
650  """
651  Get of the all possibles resolvers in this module.
652 
653  return: list of resolvers
654  rtype: list
655  """
656  return [r for r in globals().values() if cls.is_resolver(r)]
657 
658  def register_resolvers (self, resolve_order, resolvers=None, **kwargs) :
659  """
660  Registering of resolvers.
661 
662  @param resolve_order: list of resolver names to register
663  @type resolve_order: list
664 
665  @param resolvers: resolvers to register
666  @type resolvers: list
667 
668  kwargs parameters can be passed to resolvers from IPResolve constructor
669  to precize some parameters (typicaly command paths, etc.)
670  To prevent name conflicts, please use the name of resolver
671  as a prefix, like 'dns_value', 'nmblookup_path', etc.
672  """
673  self.resolvers = []
674  if not resolvers :
675  resolvers = self.get_all_resolvers()
676  else :
677  # testing if all the externals resolvers are a ResolvingCallable subclass
678  for i, candidate in enumerate(resolvers) :
679  if not self.is_resolver(candidate) :
680  log.warn("Candidate %s isn't a resolver - ignoring" % str(candidate))
681  resolvers.pop(i)
682 
683  for name in resolve_order :
684 
685  for resolver_class in resolvers :
686  resolver = resolver_class(self.networks, **kwargs)
687 
688  if name == resolver.name :
689  if resolver.validate():
690  self.resolvers.append(resolver)

750  @type target: list

789  @classmethod

791  networks = value.split()

799  def parse(self, value):
800  if not self.check_str_format(value) :
801  log.warning("Preferred network not set, using default value: %s/%s" % self.default_network[0])
802  return self.default_network
803  else :
