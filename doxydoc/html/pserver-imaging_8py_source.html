89  def test_01registerPackageServer(self):
90  """
91  Check package server registration
92  """
93  # We shouldn't have a default boot menu, because the package server is
94  # not registered
95  default = '/var/lib/pulse2/imaging/bootmenus/default'
96  self.assertFalse(os.path.exists(default))
97  ret = os.system('pulse2-package-server-register-imaging -n \"Pulse 2 imaging\" -m https://mmc:s3cr3t@localhost:7080')
98  self.assertEqual(0, ret)
99  self.assertEqual([True, True], MMCAGENT.imaging.linkImagingServerToLocation('UUID1', 'UUID1', 'root'))
100  # Restart package server and wait a bit
101  os.system('/etc/init.d/pulse2-package-server restart')
102  sleep(5)
103  # We should have a default boot menu
104  self.assertTrue(os.path.exists(default))
