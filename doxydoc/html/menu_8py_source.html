38 def isMenuStructure(menu):
39  """
40  @return: True if the given object is a menu structure
41  @rtype: bool
42  """
43  ret = True
44  logger = logging.getLogger('imaging')
45  if type(menu) == dict:
46  for k in ['message',
47  'default_item',
48  'default_item_WOL',
49  'timeout',
50  'background_uri',
51  'bootservices',
52  'images',
53  'bootcli',
54  'disklesscli',
55  'hidden_menu',
56  'dont_check_disk_size',
57  'ethercard']:
58 
59  if not k in menu:
60  logger.error("Menu is missing %s" % (k))
61  ret = False
62  break
63  else:
64  logger.error("Menu is not a dict")
65  ret = False
66  return ret

92  def cacheHostname(self, uuid, name):
93  if uuid:
94  target_folder = os.path.join(PackageServerConfig().imaging_api['base_folder'], PackageServerConfig().imaging_api['computers_folder'],uuid)
95  if os.path.isdir(target_folder):
96  self.logger.debug('Imaging menu : folder %s for client %s : It already exists !' % (target_folder, uuid))
97  return True
98  if os.path.exists(target_folder):
99  self.logger.warn('Imaging menu : folder %s for client %s : It already exists, but is not a folder !' % (target_folder, uuid))
100  return False

102  os.mkdir(target_folder)
103  self.logger.debug('Imaging menu : folder %s for client %s was created' % (target_folder,uuid))
104  except Exception, e:
105  self.logger.error('Imaging menu : I was not able to create folder %s for client %s : %s' % (target_folder, uuid, e))
106  return False
107  if name:
108  target_folder = os.path.join(target_folder,'hostname')
109  fichier = open(target_folder, "w")
110  fichier.write(name)
111  fichier.close()

246  def _applyReplacement(self, string, condition = 'global'):
247  """
248  Private func, to apply a replacement into a given string
249 
250  Some examples :
251  ##MAC:fmt## replaced by the client MAC address; ATM fmt can be:
252  - short (pure MAC addr)
253  - cisco (cisco-fmt MAC addr)
254  - linux (linux-fmt MAC addr)
255  - win (win-fmt MAC addr)
256  """
257  # list of replacements to perform
258  # a replacement is using the following structure :
259  # key 'from' : the PCRE to look for
260  # key 'to' : the replacement to perform
261  # key 'when' : when to perform the replacement (only 'global' for now)
262  replacements = [
263  ('##PULSE2_LANG##', PackageServerConfig().pxe_keymap, 'global'),
264  ('##PULSE2_PUBLIC_IP##', PackageServerConfig().public_ip, 'global'),
265  ('##PULSE2_BOOTLOADER_DIR##', self.config.imaging_api['bootloader_folder'], 'global'),
266  ('##PULSE2_TOOLS_DIR##', self.config.imaging_api['tools_folder'], 'global'),
267  ('##PULSE2_BOOTSPLASH_FILE##', self.config.imaging_api['bootsplash_file'], 'global'),
268  ('##PULSE2_DISKLESS_DIR##', self.config.imaging_api['diskless_folder'], 'global'),
269  ('##PULSE2_DAVOS_OPTS##', self.config.imaging_api['davos_options'], 'global'),
270  ('##PULSE2_DISKLESS_KERNEL##', self.config.imaging_api['diskless_kernel'], 'global'),
271  ('##PULSE2_DISKLESS_INITRD##', self.config.imaging_api['diskless_initrd'], 'global'),
272  ('##PULSE2_DISKLESS_OPTS##', ' '.join(self.diskless_opts), 'global'),
273  ('##PULSE2_KERNEL_OPTS##', ' '.join(self.kernel_opts), 'global'),
274  ('##PULSE2_MASTERS_DIR##', self.config.imaging_api['masters_folder'], 'global'),
275  ('##PULSE2_POSTINST_DIR##', self.config.imaging_api['postinst_folder'], 'global'),
276  ('##PULSE2_COMPUTERS_DIR##', self.config.imaging_api['computers_folder'], 'global'),
277  ('##PULSE2_INVENTORIES_DIR##', self.config.imaging_api['inventories_folder'], 'global'),
278  ('##PULSE2_PXE_MASK##', self.config.imaging_api['pxe_mask'], 'global'),
279  ('##PULSE2_PXE_TFTP_IP##', self.config.imaging_api['pxe_tftp_ip'], 'global'),
280  ('##PULSE2_PXE_SUBNET##', self.config.imaging_api['pxe_subnet'], 'global'),
281  ('##PULSE2_PXE_GATEWAY##', self.config.imaging_api['pxe_gateway'], 'global'),
282  ('##PULSE2_PXE_DEBUG##', self.config.imaging_api['pxe_debug'], 'global'),
283  ('##PULSE2_PXE_TIME_REBOOT##', self.config.imaging_api['pxe_time'], 'global'),
284  ('##PULSE2_PXE_XML##', self.config.imaging_api['pxe_xml'], 'global'),
285  ('##PULSE2_BASE_DIR##', self.config.imaging_api['base_folder'], 'global'),
286  ('##PULSE2_REVO_RAW##', self.rawmode, 'global')
287  ]
288  if self.mac:
289  replacements.append(
290  ('##MAC##',
291  pulse2.utils.reduceMACAddress(self.mac),
292  'global'))
293 
294  output = string
295  for replacement in replacements:
296  f, t, w = replacement
297  if w == condition:
298  output = re.sub(f, t, output)
299  return output

305  """

316  def delete_diacritics(self, s) :
317  """
318  Delete accent marks.
319 
320  @param s: string to clean
321  @type s: unicode
322  @return: cleaned string
323  @rtype: unicode
324  """
325  _reptable = self._fill_reptable()
326  if isinstance(s, str):

333  def buildMenu(self):
334  """
335  @return: the SYSLINUX boot menu as a string encoded using CP-437
336  @rtype: str
337  """
338 
339  # takes global items, one by one
340  buf = '# Auto-generated by Pulse 2 Imaging Server on %s \n\n' % time.asctime()
341  buf += 'UI vesamenu.c32\n'
342  buf += 'PROMPT 0\n'
343 
344  # the key mapping
345  buf += 'KBDMAP %s.ktl\n' % PackageServerConfig().pxe_keymap
346 
347  # PXE Password entry if pxe_password is set
348  if PackageServerConfig().pxe_password != '':
349  buf += 'MENU MASTER PASSWD %s\n' % PackageServerConfig().pxe_password
350 
351  # the menu timeout : warning, 0 means "do not wait !"
352  if self.timeout == '0':
353  buf += 'TIMEOUT 1\n'
354  else:
355  buf += 'TIMEOUT %d\n' % (int(self.timeout) * 10)
356 
357  # the menu default item
358  buf += 'DEFAULT %s\n' % self.default_item
359 
360  # the menu splash image
361  if self.splashimage:
362  buf += self._applyReplacement('MENU BACKGROUND %s\n' % self.splashimage)
363 
364  # the menu setup
365  buf += 'MENU WIDTH 78\n'
366  buf += 'MENU MARGIN 4\n'
367  buf += 'MENU ROWS 10\n'
368  buf += 'MENU VSHIFT 10\n'
369  buf += 'MENU TIMEOUTROW 18\n'
370  buf += 'MENU TABMSGROW 16\n'
371  buf += 'MENU CMDLINEROW 16\n'
372  buf += 'MENU HELPMSGROW 21\n'
373  buf += 'MENU HELPMSGENDROW 29\n'
374  buf += 'MENU TITLE %s\n' % self.hostname
375  # do we hide the menu ? Splash screen will still be displayed
376  if self.hidden:
377  buf += 'MENU HIDDEN\n'
378 
379  # then write items
380  indices = self.menuitems.keys()
381  indices.sort()
382  for i in indices:
383  output = self._applyReplacement(self.menuitems[i].getEntry())
384  buf += '\n'
385  buf += output
386  if PackageServerConfig().pxe_password != '' and not 'continue' in self.menuitems[i].getEntry():
387  buf += 'MENU PASSWD\n'
388 
389  assert(type(buf) == unicode)
390 
391  # Clean brazilian characters who are not compatible
392  # with MS-DOS encoding by delete accent marks
393  buf = self.delete_diacritics(buf)
394 
395  # Encode menu using code page 437 (MS-DOS) encoding
396  buf = buf.encode('cp437')
397  return buf

414  except Exception, e:

437  self.logger.debug('Successfully wrote boot menu for unregistered computers into file %s' % filename)
438  except Exception, e:
