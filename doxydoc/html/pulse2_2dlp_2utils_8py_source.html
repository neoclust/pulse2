43 def download_file(url, dest, retries):
44  try:
45  res = urllib2.urlopen(url)
46  path, filename = url.rsplit('/', 1)
47  blocksize = 8192
48  with open(os.path.join(dest, filename), 'wb') as f:
49  while True:
50  chunk = res.read(blocksize)
51  if not chunk:
52  break
53  f.write(chunk)
54  return True
55  except:
56  if retries > 0:
57  log("Failed to download file. Retrying...", logging.WARNING)
58  time.sleep(1)
59  return download_file(url, dest, retries - 1)
60  else:
61  log("Failed to download file:\n", logging.ERROR, True)
62  raise

65 def make_zipfile(output_filename, source_dir):
66  relroot = os.path.abspath(source_dir)
67  with zipfile.ZipFile(output_filename, "w", zipfile.ZIP_DEFLATED) as zip:
68  for root, dirs, files in os.walk(source_dir):
69  # add directory (needed for empty dirs)
70  zip.write(root, os.path.relpath(root, relroot))
71  for file in files:
72  filename = os.path.join(root, file)
73  if os.path.isfile(filename): # regular files only
74  arcname = os.path.join(os.path.relpath(root, relroot), file)
75  zip.write(filename, arcname)
