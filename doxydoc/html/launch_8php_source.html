58 function getChangedParams($post) {
59  $old_params = getOldParams($post);
60  $changed_params = array();
61  foreach ($old_params as $param) {
62  if ($post['old_' . $param] != $post[$param]) $changed_params[] = $param;
63  }
64  return $changed_params;

73 function start_a_command($proxy = array()) {
74  if ($_POST['editConvergence']) {
75  $changed_params = getChangedParams($_POST);
76  if ($changed_params == array('active')) print "We have to edit command....";
77  }
78  $error = "";
79  if (!check_date($_POST)) {

84  new NotifyWidgetFailure($error);
85  complete_post();

89  }

91  exit;

93  // Vars seeding
94  $post = $_POST;
95  $from = $post['from'];
96  $path = explode('|', $from);
97  $module = $path[0];
98  $submod = $path[1];
99  $page = $path[2];
100  $params = array();
101  foreach (array('start_script', 'clean_on_success', 'do_reboot', 'do_wol', 'next_connection_delay', 'max_connection_attempt', 'do_inventory', 'ltitle', 'parameters', 'papi', 'maxbw', 'deployment_intervals', 'max_clients_per_proxy', 'launchAction') as $param) {
102  $params[$param] = $post[$param];
103  }
104  $halt_to = array();
105  foreach ($post as $p => $v) {
106  if (preg_match('/^issue_halt_to_/', $p)) {
107  $p = preg_replace('/^issue_halt_to_/', '', $p);
108  if ($v == 'on') {
109  $halt_to[] = $p;
110  }
111  }
112  }
113  $params['issue_halt_to'] = $halt_to;
114  $p_api = new ServerAPI();
115  $p_api->fromURI($post['papi']);
116 
117  foreach (array('start_date', 'end_date') as $param) {
118  if ($post[$param] == _T("now", "msc")) {
119  $params[$param] = "0000-00-00 00:00:00";
120  } elseif ($post[$param] == _T("never", "msc")) {
121  $params[$param] = "0000-00-00 00:00:00";
122  } else {
123  $params[$param] = $post[$param];
124  }
125  }
126 
127  $pid = $post['pid'];
128  $mode = $post['copy_mode'];
129 
130  if (isset($post['uuid']) && $post['uuid']) { // command on a single target
131  $hostname = $post['hostname'];
132  $uuid = $post['uuid'];
133  $target = array($uuid);
134  $tab = 'tablogs';
135  /* record new command */
136  $id = add_command_api($pid, $target, $params, $p_api, $mode, NULL);
137  if (!isXMLRPCError()) {
138  scheduler_start_these_commands('', array($id));
139  /* then redirect to the logs page */
140  header("Location: " . urlStrRedirect("msc/logs/viewLogs", array('tab' => $tab, 'uuid' => $uuid, 'hostname' => $hostname, 'cmd_id' => $id)));
141  exit;
142  } else {
143  /* Return to the launch tab, the backtrace will be displayed */
144  header("Location: " . urlStrRedirect("msc/logs/viewLogs", array('tab' => 'tablaunch', 'uuid' => $uuid, 'hostname' => $hostname)));
145  exit;
146  }
147  } else { # command on a whole group
148  $gid = $post['gid'];
149  $tab = 'grouptablogs';
150  // record new command
151  // given a proxy list and a proxy style, we now have to build or proxy chain
152  // target structure is an dict using the following stucture: "priority" => array(proxies)
153 
154  $ordered_proxies = array();
155  if ($_POST['proxy_mode'] == 'multiple') { // first case: split mode; every proxy got the same priority (1 in our case)
156  foreach ($proxy as $p) {
157  array_push($ordered_proxies, array('uuid' => $p, 'priority' => 1, 'max_clients' => $_POST['max_clients_per_proxy']));
158  }
159  $params['proxy_mode'] = 'split';
160  } elseif ($_POST['proxy_mode'] == 'single') { // second case: queue mode; one priority level per proxy, starting at 1
161  $current_priority = 1;
162  foreach ($proxy as $p) {
163  array_push($ordered_proxies, array('uuid' => $p, 'priority' => $current_priority, 'max_clients' => $_POST['max_clients_per_proxy']));
164  $current_priority += 1;
165  }
166  $params['proxy_mode'] = 'queue';
167  }
168 
169  if (quick_get('convergence')) {
170  $active = ($_POST['active'] == 'on') ? 1 : 0;
171  $cmd_type = 2; // Convergence command type
172  if (quick_get('editConvergence')) {
173  /* Stop command */
174  $cmd_id = xmlrpc_get_convergence_command_id($gid, $p_api, $pid);
175  stop_command($cmd_id);
176  /* Set end date of this command to now(), don't touch to start date */
177  $start_date = _get_command_start_date($cmd_id);
178  extend_command($cmd_id, $start_date, date("Y-m-d H:i:s"));
179  /* Create new command */
180  $deploy_group_id = xmlrpc_get_deploy_group_id($gid, $p_api, $pid);
181  $command_id = add_command_api($pid, NULL, $params, $p_api, $mode, $deploy_group_id, $ordered_proxies, $cmd_type);
182  if (!$active) {
183  // If this convergence is not active, expire this command
184  $start_date = _get_command_start_date($command_id);
185  extend_command($command_id, $start_date, date("Y-m-d H:i:s"));
186  }
187  /* Update convergence DB */
188  $updated_datas = array(
189  'active' => $active,
190  'commandId' => intval($command_id),
191  'cmdPhases' => $params,
192  );
193  xmlrpc_edit_convergence_datas($gid, $p_api, $pid, $updated_datas);
194  }
195  else {
196  /* Create convergence */
197  // create sub-groups
198  $group = new Group($gid, True);
199  $package = to_package(getPackageDetails($p_api, $pid));
200 
201  $convergence_groups = $group->createConvergenceGroups($package);
202 
203  $deploy_group_id = $convergence_groups['deploy_group_id'];
204  $done_group_id = $convergence_groups['done_group_id'];
205 
206  // Add command on sub-group
207  $command_id = add_command_api($pid, NULL, $params, $p_api, $mode, $deploy_group_id, $ordered_proxies, $cmd_type);
208  if (!$active) {
209  // If this convergence is not active, expire this command
210  $start_date = _get_command_start_date($command_id);
211  extend_command($command_id, $start_date, date("Y-m-d H:i:s"));
212  }
213 
214  // feed convergence db
215  xmlrpc_add_convergence_datas($gid, $deploy_group_id, $done_group_id, $pid, $p_api, intval($command_id), $active, $params);
216  }
217  header("Location: " . urlStrRedirect("base/computers/groupmsctabs", array('gid' => $gid)));
218  exit;
219  }
220  else {
221  $id = add_command_api($pid, NULL, $params, $p_api, $mode, $gid, $ordered_proxies);
222  scheduler_start_these_commands('', array($id));
223  // then redirect to the logs page
224  header("Location: " . urlStrRedirect("msc/logs/viewLogs", array('tab'=>$tab, 'gid'=>$gid, 'cmd_id'=>$id, 'proxy' => $proxy)));
225  exit;
226  }
227  }

238 function check_date($post) {
239  $start = "0000-00-00 00:00:00";
240  $end = "0000-00-00 00:00:00";
241  $now = getdate();
242  $now = $now['year'] . '-' . $now['mon'] . '-' . $now['mday'] . ' ' . $now['hours'] . ':' . $now['minutes'] . ':' . $now['seconds'];
243  if ($post['start_date'] != _T("now", "msc") && $post['start_date'] != "") {
244  $start = $post['start_date'];
245  }
246  if ($post['end_date'] != _T("never", "msc") && $post['end_date'] != "") {
247  $end = $post['end_date'];
248  }
249  if ($end == "0000-00-00 00:00:00") { # never end
250  return True;
251  }
252  if ($start == "0000-00-00 00:00:00" and $end != "0000-00-00 00:00:00") {
253  if (!check_for_real($now, $end)) {
254  return False;
255  } # start now, but finish in the past
256  }
257  return check_for_real($start, $end);
258 }

260 function check_for_real($s, $e) {
261  $start = preg_split("/[ :-]/", $s);
262  $end = preg_split("/[ :-]/", $e);
263 
264  for ($i = 0; $i < 6; $i++) {
265  if ($start[$i] > $end[$i]) {
266  return False;
267  } else if ($start[$i] < $end[$i]) {
268  return True;
269  }
270  }
271  return False;
272 }
