115  def run_proxymethod(self, launcher, id, name, args, from_dlp):
116  """
117  Calls the proxy method.
118 
119  This call is invoked by proxy processing the parsing results
120  from the launcher or download provider.
121  Invoked method tagged as @launcher_proxymethod
122 
123  @param launcher: launcher which calls this method
124  @type launcher: str
125 
126  @param id: commands_on_host id
127  @type id: int
128 
129  @param args: argumets of called method
130  @type args: list
131 
132  @param from_dlp: if True, response is coming from DLP
133  @type from_dlp: bool
134  """
135  circuit = self.get(id)
136  if circuit :
137  return self._run_proxymethod(launcher, id, name, args, circuit)
138  else :
139  self.logger.info("probably recurrent phase or stopped circuit #%s" % (id))
140  # Recurrent phase parsing which does not exists in the container
141  # The circuit is created out of container
142  circuit = Circuit(id, self.installed_phases, self.config, from_dlp)
143  if not circuit :
144  self.logger.warn("Circuit #%s: not found" % id)
145  return False
146  circuit.install_dispatcher(self)
147  d = circuit.setup(True)
148  @d.addCallback
149  def _setup(result):
150  dth = deferToThread(circuit.run)
151  return dth
152  @d.addCallback
153  def _post_setup(result):
154  if not circuit.running_phase :
155  self.logger.warn("Method <%s> call for current phase is not valid" % name)
156  return False
157  if from_dlp :
158  valid_method_name = get_dlp_method(circuit.running_phase.name)
159 
160  if valid_method_name != name :
161  self.logger.warn("Recurrent phase %s ignored" % name)
162  return False
163  dps = maybeDeferred(self._run_proxymethod,
164  launcher,
165  id,
166  name,
167  args,
168  circuit)
169  @dps.addCallback
170  def result_proxy(res):
171  return True
172 
173  @dps.addErrback
174  def failed_proxy(failure):
175  self.logger.warn("Proxymethod execution failed: %s" % failure)
176  return False
177  return dps
178 
179 
180  @d.addErrback
181  def _eb(result):
182  self.logger.warn("Recurrent phase result parsing failed: %s" % result)
183  return False
184  return d

187  def _run_proxymethod(self, launcher, id, name, args, circuit=None):
188  """
189  Calls the proxy method.
190 
191  @param launcher: launcher which calls this method
192  @type launcher: str
193 
194  @param id: commands_on_host id
195  @type id: int
196 
197  @param args: argumets of called method
198  @type args: list
199 
200  @param circuit: Recurrent circuit
201  @type circuit: Circuit
202  """
203 
204  if hasattr(circuit.running_phase, "proxy_methods"):
205  px_dict = getattr(circuit.running_phase, "proxy_methods")
206  if name in px_dict :
207  method_name = px_dict[name].__name__
208  method = px_dict[name]
209  if not hasattr(circuit.running_phase, method_name):
210  self.logger.debug("Proxymethod %s not found, call aborted" % name)
211  return False
212 
213  self.logger.debug("Incoming result from launcher <%s>,executing method <%s> from phase <%s>" %
214  (launcher, method_name, circuit.running_phase.name))
215  if len(args) == 3:
216  exitcode, _stdout, _stderr = args
217  stdout = unicode(b64decode(_stdout),'utf-8', 'strict')
218  stderr = unicode(b64decode(_stderr),'utf-8', 'strict')
219  result = method(circuit.running_phase, (exitcode, stdout, stderr))
220  else:
221  result = method(circuit.running_phase, args)
222 
223  circuit.phase_process(result)
224  return True
225  return False

257  def start_all(self, ids, truncate=False):
258  """
259  A starting point of the workflow.
260 
261  @param ids: list of ids (commands_on_host) to start.
262  @type ids: list
263 
264  """
265  d1 = Deferred()
266 
267  banned = self.bundles.get_banned_cohs()
268  ids = [id for id in ids if id not in banned]
269 
270  already_initialized_circuits = [c for c in self._circuits
271  if c.id in ids
272  ]
273  # already initialized circuits will be started
274  aicd = maybeDeferred(self._circuits_activate,
275  already_initialized_circuits)
276  aicd.addCallback(self._run_all)
277  aicd.addErrback(self._start_failed)
278 
279  already_initialized_ids = [c.id for c in already_initialized_circuits]
280 
281  new_ids = [id for id in ids if not id in already_initialized_ids]
282 
283  if truncate and len(new_ids) > self.free_slots :
284 
285  self.logger.info("Number of new circuits was truncated from %d to %d" %
286  (len(new_ids), self.free_slots))
287  new_ids = new_ids[:self.free_slots]
288  self.logger.info("Remaining circuits will be requested soon")
289 
290  if len(new_ids) > 0 :
291  d1 = self._setup_all(new_ids)
292 
293  d1.addCallback(self._assign_launcher)
294  d1.addCallback(self._class_bundles)
295  d1.addCallback(self._class_all)
296  d1.addCallback(self._revolve_all)
297  d1.addCallback(self._run_all)
298  d1.addErrback(self._start_failed)
299  return d1
300  else:
301  succeed(True)

504  def _class_all(self, incoming_circuits):
505  """
506  Classing of all incoming circuits.
507 
508  Based on statistics of running circuits (see _select_balanced),
509  this treatement ensures the balanced execution of favorized circuits
510  and remaining circuits temporarily saves into a queue.
511 
512  @param incoming_circuits: circuits to class
513  @type incoming_circuits: deferred list
514  """
515  new_circuits = []
516  banned = self.bundles.get_banned_cohs()
517  circuits_to_start = []
518  for circuit in incoming_circuits :
519  circuit.status = CC_STATUS.WAITING
520  new_circuits.append(circuit)
521 
522  grouped = self._select_balanced(new_circuits)
523 
524  for group, total in grouped.items() :
525  circuits_to_run = []
526 
527  count = 0
528  for circuit in [c for c in new_circuits if c.network_address==group] :
529 
530  # bundle banned excluding
531  if circuit.id in banned:
532  self.logger.info("Circuit #%s is part of bundle, still waiting" % circuit.id)
533  continue
534  if not circuit.cohq.cmd.inDeploymentInterval():
535  circuit.release()
536  continue
537  if self.contains_running_target(circuit):
538  # if another deployment running on the same machine
539  continue
540  if circuit.network_address == group :
541  if count == total :
542  break
543  # include a new circuit to running container
544  circuit.status = CC_STATUS.ACTIVE
545  self.started_track.add([circuit.id])
546  circuits_to_run.append(circuit)
547  new_circuits.remove(circuit)
548  count += 1
549 
550 
551  circuits_to_start.extend(circuits_to_run)
552 
553 
554  # rest of new circuits in queue
555  new_circuits = [c for c in new_circuits if c not in self.waiting_circuits]
556 
557  self._circuits.extend(new_circuits)
558  self.logger.info("Circuits stats: running: %d waiting: %d" %
559  (len(self.circuits), len(self.waiting_circuits)))
560 
561  return circuits_to_start

563  def _revolve_all(self, circuits_to_start):
564  """ Randomizing the deployment orders by group """
565 
566  final_length = len(circuits_to_start)
567  circuits_by_groups = {}
568  for group in self.groups :
569  content = [c for c in circuits_to_start if c.network_address==group]
570  random.shuffle(content)
571 
572  circuits_by_groups[group] = content
573 
574  circuits = []
575  while True :
576  if len(circuits) == final_length:
577  break
578  for group in self.groups :
579  try :
580  c = circuits_by_groups[group].pop(0)
581  circuits.append(c)
582  except IndexError:
583  continue
584 
585  return circuits

588  def _select_balanced(self, new_circuits):
589  """
590  Equalizing the circuits by groups.
591 
592  Based on statistics of running circuits, this routine calculates
593  the number of new circuits to execute on slihghtests groups.
594  Goal is the equalized load balancing to avoid saturate some networks
595  more than others.
596  Final number of new_circuits, together with running circuits,
597  covers the maximum of running circuits (max_slots).
598 
599  @param new_circuits: new circuits to equalize
600  @type new_circuits: list
601 
602  @return: number of circuits per existing groups.
603  @rtype: dict
604  """
605 
606  # sorted statistics from circuits to add
607  new_grps_stat = self._analyze_groups(new_circuits)
608  # sorted statistics from running circuits
609  running_grps_stat = self._analyze_groups(self.circuits)
610 
611  to_add = None
612 
613  if len(self.circuits) < self.max_slots :
614  # number of new circuits to add
615  free_slots = self.max_slots - len(self.circuits)
616 
617  if len(self.circuits) + len(new_circuits) <= self.max_slots :
618  return new_grps_stat
619  else :
620 
621  to_add = dict((group, 0) for group in self.groups)
622 
623  zero_blacklist = []
624  while True :
625 
626  masked = dict((group, value + to_add[group])
627  for (group, value) in running_grps_stat.items()
628  if group not in zero_blacklist)
629  # group having minimum circuits
630  min_key = min(masked, key=masked.get)
631  if new_grps_stat[min_key] > 0 :
632  to_add[min_key] += 1
633  new_grps_stat[min_key] -= 1
634  else :
635  zero_blacklist.append(min_key)
636 
637  if sum(to_add.values()) == free_slots :
638  break
639  else :
640  return dict((g, 0) for g in self.groups)
641 
642  return to_add

667  def _get_next_waiting(self, circuits, slightest_network):
668  """
669  Selects a next candidate to launch.
670 
671  @param circuits: proposed circuits
672  @type circuits: list
673 
674  @param slightest_network: group with the minimum running circuits
675  @type slightest_network: str
676 
677  @return: next circuit to start
678  @rtype: Circuit
679 
680  """
681  if len(circuits) > 0 :
682  ids = [c.id for c in circuits if c.network_address==slightest_network]
683 
684  # queued bundled circuits to exclude
685  banned = self.bundles.get_banned_cohs()
686  for id in ids :
687  if id in banned :
688  continue
689 
690  # do not start a stopped circuit
691  if id in self.stopped_track:
692  circuit = self.get(id)
693  if circuit :
694  circuit.release()
695  continue
696 
697  circuit = self.get(id)
698  if circuit :
699  if self.contains_running_target(circuit):
700  # if another deployment running on the same machine
701  continue
702 
703  self.logger.info("Circuit #%s: (group %s) is going to start" %
704  (circuit.id, slightest_network))
705  circuit.status = CC_STATUS.ACTIVE
706  return circuit
707  return None

709  def launch_next_waiting(self):
710  """
711  Launch the next candidate to start.
712 
713  The choice of next circuit is based on statistics of running circuits.
714  Like on start, goal is selecting a circuit destinated to a computer
715  which is located at least saturated network.
716  """
717  try:
718  running = self._analyze_groups(self.circuits)
719  remaining = self._analyze_groups(self.get_valid_waitings())
720 
721  for _ in xrange(self.nbr_groups):
722  # the least saturated network
723  slightest_group = min(running, key=running.get)
724  if remaining[slightest_group] > 0 :
725  # waiting circuits not processed yet
726  unprocessed_circuits = self.get_unprocessed_waitings()
727  # waiting circuits already processed (recycling of failed attempts)
728  already_treated_circuits = self.get_valid_waitings()
729 
730  for circuits in [unprocessed_circuits, already_treated_circuits]:
731  if self.has_free_slots():
732  circuit = self._get_next_waiting(circuits, slightest_group)
733  if circuit :
734  circuit.run()
735  return True
736  else:
737  return False
738 
739  else :
740  if slightest_group in running :
741  del running[slightest_group]
742  # if not a candidate in waiting circuits, skip on next group
743  continue
