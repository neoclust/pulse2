24 """ A XMLRPC Client
25 
26 Arguments must follow this convention:
27  - args are splitted over ';' (we call it an arg)
28 
29  => use --func myfunc --args "a;b;c" to request func('a', 'b', 'c')
30 
31  - in one arg, array/dict component are splitted over '|' (we call it a token)
32 
33  => use --func myfunc --args "a|b" to request func(['a', 'b'])
34 
35  - if an token contain '=', it is interpreted as a tuple
36 
37  => use --func myfunc --args "a=b|c=d" to request func(['a': 'b', 'c': 'd'])
38 
39  - if an token contain ',', it is interpreted as a list
40 
41  => use --func myfunc --args "a,b,c,d" to request func('a', 'b', 'c', 'd')
42 
43  => use --func myfunc --args "a,b|c,d" to request func(['a', 'b'], ['c', 'd'])
44 """

77 # parse cli args
78 def parseCliArgs(config):
79  args=[]
80  method = config["func"]
81  if config["args"]:
82  args = config["args"].split(';')
83 
84  parsedargs = []
85  for arg in args: # parse args
86  tokenlist = arg.split('|') # split arrays args
87  for token in tokenlist: # iterate over array content
88  if token.count('=') == 1: # found a dict token
89  (key, val) = token.split('=') # process it
90  try:
91  items = []
92  items[key] = val
93  except:
94  items = {}
95  items = {key: val}
96  elif token.count(',') > 0: # found a list token
97  try:
98  items += [token.split(',')]
99  except:
100  items = [token.split(',')]
101  elif token.count('~') == 1: # found a number
102  try:
103  items += int(token.split('~')[1])
104  except:
105  items = [int(token.split('~')[1])]
106  else: # found something else (simple value ?)
107  try:
108  items += token
109  except:
110  items = [token]
111 
112  if type(items) == type({}):
113  parsedargs.append(items)
114  elif type(items) == type([]):
115  parsedargs += items
116  else:
117  parsedargs += items
118  del(items)
119 
120  return (method, parsedargs)
