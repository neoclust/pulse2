102 function displayLogs($params) {
103 
104  $result1 = xmlCall('msc.displayLogs', array($params));
105 
106  if (!empty($_GET['create_group'])){
107 
108  // Small hack to include all pages
109  $params['min'] = 0;
110  $params['max'] = 9999999;
111 
112  $result = xmlCall('msc.displayLogs', array($params));
113 
114  require_once("modules/dyngroup/includes/dyngroup.php");
115  require_once("modules/dyngroup/includes/xmlrpc.php");
116 
117 
118  // Group name
119  $cmd_name = $result[1][0][0]['title'];
120  $groupname = sprintf (_T("%s deployment subgroup - %s", "msc"), $cmd_name, date("Y-m-d H:i:s"));
121 
122  // Creating group with displayLogs result filter
123  $coh_ids = array();
124  foreach ($result[1] as $entry){
125  $coh_ids[] = $entry[3]['id'];
126  }
127 
128  $targets = get_targets_for_coh($coh_ids);
129 
130  // Adding group members
131  $groupmembers = array();
132  foreach ($targets as $target) {
133  $uuid = $target['target_uuid'];
134  $cn = $target['target_name'];
135  $groupmembers["$uuid##$cn"] = array('hostname' => $cn, 'uuid' => $uuid);
136  }
137 
138  $group = new Group();
139  $group->create($groupname, False);
140  $group->addMembers($groupmembers);
141 
142  $link = urlStrRedirect("base/computers/display", array('gid'=>$group->id));
143  new NotifyWidgetSuccess(sprintf(_T('Deploy subgroup created successfully. <br/><a href="%s" target="_blank">View group</a>', "msc"), $link));
144  }
145 
146  return $result1;
147 }
