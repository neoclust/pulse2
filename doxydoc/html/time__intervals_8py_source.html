112  self.segments = self._merge_r(TimeSegment(segment.start, segment.end))
113  else:

117  def _merge_r(self, new_segment):
118  """ Merges new_segment in self.segments """
119  if new_segment == None:
120  return self.segments
121  if len(self.segments) == 0: # nothing to be merged to, return seed
122  return [new_segment]
123  # try to merge
124  poped_segment = self.segments.pop()
125  merged_segment = _merge(poped_segment, new_segment)
126  if len(merged_segment) == 1: # got one new segment after merging => ovelap
127  return self._merge_r(merged_segment[0])
128  if len(merged_segment) == 2: # got 2 segments after merging => no overlap
129  # as there was no overlap, merged segment contains poped_segment and new_segment
130  # we attempts to merge new_segment with remaining values from segments,
131  # and returns poped_segment appart
132  ret = self._merge_r(new_segment)
133  ret.append(poped_segment)
134  return ret

136 def _merge(s1, s2):
137  """ Merges 2 TimeSegment() together
138  May return 1 array of 1 segment (in case of merge)
139  or 1 array of 2 segments (no merge occured)
140  in this case segments are in the same order as given
141  """
142  # start point are the same, return the bigger one
143  if s1.start == s2.start:
144  if s1.end >= s2.end:
145  return [s1]
146  else:
147  return [s2]
148  # end point are the same, return the bigger one
149  elif s1.end == s2.end:
150  if s1.start <= s2.start:
151  return [s1]
152  else:
153  return [s2]
154  # things are getting complicated, now s1 starts before s2
155  elif s1.start < s2.start: # s1 "starts" before s2 "starts
156  if s1.end < s2.start:
157  return [s1, s2] # s1 "finished" before s2 "starts", merge is s1, s2
158  elif s1.end < s2.end:
159  return [TimeSegment(s1.start, s2.end)] # s1 "finished" before s2 "end", merge is s1.start, s2.end
160  elif s1.end >= s2.end:
161  return [s1] # s1 "finished" after s2 "end", merge is s1
162  # last case: s2 starts before s1
163  elif s2.start < s1.start: # s2 "starts" before s1 "starts
164  if s2.end < s1.start:
165  return [s2, s1] # s2 "finished" before s1 "starts", merge is s1, s2
166  elif s2.end < s1.end:
167  return [TimeSegment(s2.start, s1.end)] # s2 "finished" before s1 "end", merge is s2.start, s1.end
168  elif s2.end >= s1.end:
169  return [s2] # s2 "finished" after s1 "end", merge is s2

171 def string2timeinterval(string):
172  """ handle conversion from string to timeinterval """
173  if not string:
174  return None
175  tp = TimeInterval()
176  for segment in string.split(','):
177  split = segment.split('-')
178  if len(split) == 2:
179  (start, end) = map(TimePoint, split)
180  if not (start.valid and end.valid):
181  return None
182  tp.add(TimeSegment(start, end))
183  else:
184  return None
185  return tp
