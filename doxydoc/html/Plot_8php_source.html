190  private function computeGraphArea() {
191  $titleUnpaddedBottom = $this->imageArea->y1 + $this->titleHeight + $this->titlePadding->top + $this->titlePadding->bottom;
192  $graphArea = null;
193  if ($this->hasCaption) {
194  $graphUnpaddedRight = $this->imageArea->x1 + ($this->imageArea->x2 - $this->imageArea->x1) * $this->graphCaptionRatio
195  + $this->graphPadding->left + $this->graphPadding->right;
196  $graphArea = new Rectangle(
197  $this->imageArea->x1,
198  $titleUnpaddedBottom,
199  $graphUnpaddedRight - 1,
200  $this->imageArea->y2
201  );
202  } else {
203  $graphArea = new Rectangle(
204  $this->imageArea->x1,
205  $titleUnpaddedBottom,
206  $this->imageArea->x2,
207  $this->imageArea->y2
208  );
209  }
210  $this->graphArea = $graphArea->getPaddedRectangle($this->graphPadding);
211  }

216  private function computeCaptionArea() {
217  $graphUnpaddedRight = $this->imageArea->x1 + ($this->imageArea->x2 - $this->imageArea->x1) * $this->graphCaptionRatio
218  + $this->graphPadding->left + $this->graphPadding->right;
219  $titleUnpaddedBottom = $this->imageArea->y1 + $this->titleHeight + $this->titlePadding->top + $this->titlePadding->bottom;
220  $captionArea = new Rectangle(
221  $graphUnpaddedRight,
222  $titleUnpaddedBottom,
223  $this->imageArea->x2,
224  $this->imageArea->y2
225  );
226  $this->captionArea = $captionArea->getPaddedRectangle($this->captionPadding);
227  }
