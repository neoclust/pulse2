416  def completeNomenclatureLabel(self, objs):
417  if type(objs) != list and type(objs) != tuple:
418  objs = [objs]
419  if len(objs) == 0:
420  return
421  className = str(objs[0].__class__).split("'")[1].split('.')[-1]
422  nomenclatures = []
423  if className in self.fk_nomenclatures:
424  for i in self.fk_nomenclatures[className]:
425  nomenclatures.append([i, i.replace('fk_', ''), self.nomenclatures[self.fk_nomenclatures[className][i]]])
426  for obj in objs:
427  for fk, field, value in nomenclatures:
428  fk_val = getattr(obj, fk)
429  if fk == field:
430  field = '%s_value' % field
431  if fk_val in value:
432  setattr(obj, field, value[fk_val])
433  else:
434  self.logger.warn("nomenclature is missing for %s field %s (value = %s)" % (str(obj), field, str(fk_val)))
435  setattr(obj, field, "%s:nomenclature does not exists." % (P2ERR.ERR_MISSING_NOMENCLATURE))
