66  def doProvisioning(self, authtoken):
67  userentry = authtoken.getInfos()[1]
68  uid = authtoken.getLogin()
69  if self.config.profileAttr and self.config.profilesEntity:
70  try:
71  profile = userentry[self.config.profileAttr][0].lower()
72  except KeyError:
73  self.logger.info("No profile information for user '%s' in attribute %s" % (uid, self.config.profileAttr))
74  profile = ''
75  profile = profile.strip()
76  try:
77  entities = self.config.profilesEntity[profile].split()
78  except KeyError:
79  if "default" in self.config.profilesEntity:
80  entities = self.config.profilesEntity["default"].split()
81  self.logger.info("Set the default profile to user.")
82  profile = 'default'
83  else:
84  self.logger.info("No entity defined in configuration file for profile '%s'" % profile)
85  self.logger.info("Setting user's entity to empty")
86  entities = []
87  if profile and entities:
88  tmp = []
89  for entity in entities:
90  if entity.startswith('%') and entity.endswith('%'):
91  attr = entity.strip('%')
92  if attr in userentry:
93  tmp.extend(userentry[attr])
94  else:
95  self.logger.info("The user '%s' doesn't have an attribute '%s'" % (uid, attr))
96  elif entity.startswith('plugin:'):
97  plugin = entity.replace('plugin:', '')
98  searchpath = os.path.join(os.path.dirname(__file__), 'provisioning_plugins')
99  try:
100  f, p, d = imp.find_module(plugin, [searchpath])
101  mod = imp.load_module(plugin, f, p, d)
102  klass = mod.PluginEntities
103  found = klass().get(authtoken)
104  if found:
105  self.logger.info("Plugin '%s' found these entities: %s" % (plugin, found))
106  else:
107  self.logger.info("Plugin '%s' found no matching entity" % plugin)
108  tmp.extend(found)
109  except ImportError:
110  self.logger.error("The plugin '%s' can't be imported" % plugin)
111  except Exception, e:
112  self.logger.error("Error while using the plugin '%s'" % plugin)
113  self.logger.exception(e)
114  else:
115  tmp.append(entity)
116  entities = tmp[:]
117  self.logger.info("Setting user '%s' entities corresponding to user profile '%s': %s" % (uid, profile, str(entities)))
118  Inventory().setUserEntities(uid, entities)
