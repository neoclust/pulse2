144  def _parse_filter_on(self, value):
145  """
146  Parsing of customized filters.
147 
148  Returned value will be parsed as a dictionnary with list of values
149  for each filter.
150 
151  @param value: raw string
152  @type value: str
153 
154  @return: dictionnary of filters
155  @rtype: dict
156  """
157  try:
158  couples = [f.split("=") for f in value.split(" ")]
159 
160  filters = dict([(key, values.split("|")) for (key, values) in couples])
161  logging.getLogger().debug("will filter machines on %s" % (str(filters)))
162  return filters
163 
164  except Exception, e:
165  logging.getLogger().warn("Parsing on filter_on failed: %s" % str(e))
166  return None
