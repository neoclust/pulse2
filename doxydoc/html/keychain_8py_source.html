29 def get_keychain():

34  if os.path.exists('/root/.keychain/'+out[0]+'-sh'):
35  f=open('/root/.keychain/'+out[0]+'-sh', 'r')

38 
39  proc = shLaunch('env '+lines[0])
40  return proc.out
41
