79  def do_POST(self):
80  content = self.rfile.read(int(self.headers['Content-Length']))
81  resp = ''
82  from_ip = self.client_address[0]
83  deviceid = ''
84 
85  # handle compressed inventories
86  # application/x-compress is for OCS
87  # application/x-compress is for Fusion
88  if 'compress' in self.headers['Content-Type']:
89  try:
90  decomp = decompressobj()
91  content = decomp.decompress(content)
92  if decomp.unused_data:
93  self.logger.warn("The content of the request from %s seems to be bad."%(from_ip))
94  self.logger.debug("The remaining bytes are : %s"%(decomp.unused_data))
95  except Exception, e:
96  self.logger.error("Failed while decompressing the request from %s."%(from_ip))
97  self.logger.error(str(e))
98 
99  cont = [content, self.headers['Content-Type']]
100 
101  # Let's figure out a few things about this incoming XML...
102  try:
103  query = re.search(r'<QUERY>([\w-]+)</QUERY>', content).group(1)
104  except AttributeError, e:
105  self.logger.warn("Could not get any QUERY section in inventory from %s" % from_ip)
106  query = 'FAILS'
107  try:
108  if query != 'UPDATE':
109  deviceid = re.search(r'<DEVICEID>([\w.-]+)</DEVICEID>', content.decode('utf8'), re.UNICODE).group(1)
110  except AttributeError, e:
111  self.logger.warn("Could not get any DEVICEID section in inventory from %s"%(from_ip))
112  self.logger.debug("no DEVICEID in %s"%(content))
113  query = 'FAILS'
114 
115  if query == 'PROLOG':
116  self.logger.info("PROLOG received from %s (DEVICEID: %s)" % (from_ip, deviceid))
117  config = InventoryGetService().config
118  resp = '<?xml version="1.0" encoding="utf-8" ?><REPLY>'
119  for section in config.options:
120  try:
121  params = config.options[section]
122  resp += '<OPTION><NAME>%s</NAME>' % (params['name'])
123  resp_param = ""
124  for p in params['param']:
125  resp_param += '<PARAM '
126  for attr in p['param']:
127  resp_param += '%s="%s" ' % (attr[0], attr[1])
128  resp_param += '>%s</PARAM>' % (p['value'])
129  resp += resp_param + '</OPTION>'
130  except:
131  self.logger.error('please check your %s config parameter' % (section))
132  resp = resp + '<RESPONSE>SEND</RESPONSE>'
133  self.logger.debug('Inventory periodicity set to %s' % str(config.inventory_periodicity))
134  resp = resp + '<PROLOG_FREQ>'+str(config.inventory_periodicity)+'</PROLOG_FREQ>'
135  resp = resp + '</REPLY>'
136  elif query == 'UPDATE':
137  self.logger.info("UPDATE received from %s (DEVICEID: %s)" % (from_ip, deviceid))
138  resp = '<?xml version="1.0" encoding="utf-8" ?><REPLY><RESPONSE>no_update</RESPONSE></REPLY>'
139  elif query == 'INVENTORY':
140  self.logger.info("INVENTORY received from %s (DEVICEID: %s)" % (from_ip, deviceid))
141  resp = '<?xml version="1.0" encoding="utf-8" ?><REPLY><RESPONSE>no_account_update</RESPONSE></REPLY>'
142  Common().addInventory(deviceid, from_ip, cont)
143  # Forwarding the inventories to GLPI (if enabled)
144  if self.config.enable_forward or self.config.enable_forward_ocsserver:
145  self.glpi_forward(content, from_ip, query)
146 
147  self.send_response(200)
148  self.end_headers()
149  self.wfile.write(compress(resp))

182  self.logger.debug("<GlpiProxy> Match found using %s! UUID: %s" % (macaddr, str(glpi_uuid)))
183  has_known_os = hasKnownOS(glpi_uuid)
184  break
185  else:
186  self.logger.info("<GlpiProxy> Unable to resolve machine ID using MAC %s New machine?" % ', '.join(macaddresses))
187

191  # Machine found in database with real OS

193  # Machine found in database with real OS and new inventory is PXE

195  self.logger.info("<GlpiProxy> Machine %s received a new PXE inventory from %s: skipping (don't overwrite real inventory)" % (str(glpi_uuid), str(from_ip)))
196  do_forward = False
197  # Machine found in database with real OS and new inventory is real inventory
198  else:
199  self.logger.info("<GlpiProxy> Machine %s received a new inventory from %s: forwarding" % (str(glpi_uuid), str(from_ip)))
200  do_forward = True
201  # Machine found in database with PXE OS
202  else:
203  # Machine found in database with PXE OS and new inventory is PXE
204  if InventoryUtils.is_coming_from_pxe(content):
205  self.logger.info("<GlpiProxy> Machine %s received a new PXE inventory from %s: forwarding (overwrite PXE inventory)" % (str(glpi_uuid), str(from_ip)))
206  do_forward = True
207  # Machine found in database with PXE OS and new inventory real inventory
208  else:
209  self.logger.info("<GlpiProxy> Machine %s received a new inventory from %s: forwarding (overwrite PXE inventory)" % (str(glpi_uuid), str(from_ip)))
210  do_forward = True
211  # Machine is not known, forward anyway
212  else:
213 
214  if canDoInventory():
215  if InventoryUtils.is_coming_from_pxe(content):
216  self.logger.info("<GlpiProxy> PXE inventory received from %s for an unknown machine: forwarding" % str(from_ip))
217  do_forward = True
218  else:
219  self.logger.info("<GlpiProxy> Inventory received from %s for an unknown machine: forwarding" % str(from_ip))
220  do_forward = True
221  else:
222  self.logger.info("<GlpiProxy> Cannot forward inventory (operation denied)")
223 
224  # Let's forward if needed
225  if do_forward:
226 
227  # Let's fix the XML using py config scripts
228  invfix = InventoryFix(self.config, content)

238  else:
239  self.logger.info("<GlpiProxy> Forwarding query %s from %s" % (str(query), str(from_ip)))

263  def _check_in (self):
264  """
265  Find and pre-check all .py from xmlfixplugindir.
266  Checked module must have a calable function named 'xml_fix'.
267  """
268  for (path, dirs, files) in os.walk(self.config.xmlfixplugindir):
269  for filename in sorted(files):
270  pathname = os.path.join(path, filename)
271  if re.match('^.*\.py$',pathname):
272  mod_name = filename
273  py_mod = fnc = None
274  try:
275  py_mod = imp.load_source(mod_name, pathname)
276 
277  except ImportError :
278  self.logger.warn("Cannot load fixing script '%s'" % filename)
279  continue
280  except Exception, e :
281  self.logger.warn("Unable to run %s script: %s" % (filename, e))
282  continue
283 
284  if hasattr(py_mod, 'xml_fix'):
285  fnc = getattr(py_mod, 'xml_fix')
286  if hasattr(fnc, "__call__") :
287  self.fixers.append(fnc)
288  else :
289  self.logger.warn("module %s : attribute xml_fix is not a function or method" % filename)
290  else :
291  self.logger.warn("Unable to run %s script: missing xml_fix() function" % filename)

372  self.logger.debug('%s' % cont)

374  macaddresses = InventoryUtils.getMACs(content)
375  # Honestly, 00:00:00:00:00:00 can't be a good mac, trust me I'm an engineer
376  macaddresses = [x for x in macaddresses if not x == '00:00:00:00:00:00']
377  self.logger.info("MAC addresses found: %s" % ', '.join(macaddresses))
378  final_macaddr = None
379 
380  setLastFlag = True
381 
382  # GLPI case - inventory creating disabled
383  if self.config.enable_forward and len(macaddresses) > 0:
384  try :
385  for macaddr in macaddresses:
386  self.logger.debug("LightPull: Trying to associate %s to an existing GLPI machine using MAC %s" % (deviceid, macaddr))
387  glpi_machine_uuid = resolveGlpiMachineUUIDByMAC(macaddr)
388  if glpi_machine_uuid :
389  self.logger.debug("LightPull: Machine %s resolved as GLPI %s using MAC %s" % (deviceid, str(glpi_machine_uuid), macaddr))
390  AttemptToScheduler(content, glpi_machine_uuid)
391  final_macaddr = macaddr
392  break
393  else:
394  self.logger.info("LightPull: Unable to resolve %s from GLPI using MAC %s, new machine?" % (deviceid, ', '.join(macaddresses)))
395  except Exception, exc :
396  self.logger.error("LightPull: An error occurred when trying to resolve UUID from GLPI: %s" % str(exc))
397 
398  if self.config.enable_forward and not self.config.enable_forward_ocsserver:
399  return False
400 
401  # Native case - inventory handling
402  coming_from_pxe = False
403 
404  if InventoryUtils.is_coming_from_pxe(content):
405  self.logger.debug("Inventory is coming from PXE")
406  inv = Inventory()
407  inv.activate(self.config)
408  setLastFlag = not inv.isInventoried(final_macaddr)
409  coming_from_pxe = True
410 
411  try:
412  start_date = time.time()
413  threadname = threading.currentThread().getName().split("-")[1]
414  inv_data, encoding, date = '', '', strftime("%Y-%m-%d %H:%M:%S")
415  current_entity = ''
416 
417  self.logger.debug("Thread %s : starting process : %s " % (threadname, time.time()))
418  try:
419  inv_data = re.compile(r'<CONTENT>(.+)</CONTENT>', re.DOTALL).search(content).group(1)
420  except AttributeError, e:
421  # we can not work without it!
422  self.logger.warn("Could not get any CONTENT section in inventory from %s"%(from_ip))
423  return False
424 
425  try:
426  encoding = re.search(r' encoding=["\']([^"\']+)["\']', content).group(1)
427  except AttributeError, e:
428  self.logger.warn("Could not get any encoding in inventory from %s"%(from_ip))
429 
430  try:
431  date = re.compile(r'<LOGDATE>(.+)</LOGDATE>', re.DOTALL).search(inv_data).group(1)
432  except AttributeError, e:
433  # we can work without it
434  self.logger.warn("Could not get any LOGDATE section in inventory from %s"%(from_ip))
435 
436  try:
437  current_entity = re.compile(r'<TAG>(.+)</TAG>', re.DOTALL).search(content).group(1)
438  except AttributeError, e:
439  # we can work without it
440  self.logger.debug("Could not get any TAG section in inventory from %s" % from_ip)
441 
442  self.logger.debug("Thread %s : regex : %s " % (threadname, time.time()))
443  inventory = '<?xml version="1.0" encoding="%s" ?><Inventory>%s</Inventory>' % (encoding, inv_data)
444  inventory = re.sub(r'</?HISTORY>', '', inventory)
445  inventory = re.sub(r'</?DOWNLOAD>', '', inventory)
446 
447  # Let's fix the XML using py config scripts
448  invfix = InventoryFix(self.config, inventory)
449  inventory = invfix.get()
450  # Store data on the server
451  inventory = OcsMapping().parse(inventory)
452  self.logger.debug("Thread %s : parsed : %s " % (threadname, time.time()))
453  hostname = '-'.join(deviceid.split('-')[0:-6])
454  self.logger.debug("Thread %s : Original hostname : %s" % (threadname, hostname))
455  try:
456  path = Pulse2OcsserverConfigParser().hostname
457  # WARNING : no fallback if the tag does not exists....
458  if len(path) == 3:
459  if path[0] in inventory:
460  found_tag = False
461  for tag in inventory[path[0]]:
462  self.logger.debug("tag = %s" % tag)
463  if path[2][0] in tag and tag[path[2][0]] == path[2][1] and path[1] in tag:
464  found_tag = True
465  hostname = tag[path[1]]
466  if not found_tag:
467  self.logger.warn("Thread %s : Can't alter hostname for %s using tag: tag value for %s/%s when %s/%s == %s not found" % (threadname, hostname, path[0], path[1], path[0], path[2][0], path[2][1]))
468  else:
469  self.logger.warn("Thread %s : Can't find %s in inventory to alter hostname for %s" % (threadname, path[0], hostname))
470 
471  self.logger.debug("Thread %s : Final hostname, 3 components path: %s" % (threadname, hostname))
472  else:
473  hostname = inventory[path[0]][0][path[1]]
474  self.logger.debug("Thread %s : Final hostname, 2 components path: %s" % (threadname, hostname))
475  except Exception, e:
476  self.logger.exception(e)
477  self.logger.error("inventory = %s" % inventory)
478 
479  # Assign an entity to the computer
480  entities = InventoryCreator().rules.compute(inventory)
481  if entities:
482  entity = entities[0]
483  else:
484  entity = InventoryCreator().config.default_entity
485 
486  # If no rule, use the entity in TAG
487  if entity == InventoryCreator().config.default_entity and current_entity:
488  entity = current_entity
489 
490  self.logger.info("Computer %s assigned to entity %s" % (hostname, entity))
491  inventory['Entity'] = [{'Label' : entity}]
492 
493  self.logger.debug("Thread %s : prepared : %s " % (threadname, time.time()))
494  result = InventoryCreator().createNewInventory(hostname, inventory, date, setLastFlag, coming_from_pxe=coming_from_pxe, from_ip=from_ip)
495  self.logger.debug("Thread %s : done : %s " % (threadname, time.time()))
496  # TODO if ret == False : reply something else
497  end_date = time.time()
498 
499  self.logger.info("Injected inventory for %s in %s seconds" % (hostname, end_date - start_date))
500 
501  ret = None
502  if isinstance(result, list) and len(result) == 2 :
503  # disabling light pull on GLPI mode when Pulse2 inventory creator is enabled
504  if not self.config.enable_forward or self.config.enable_forward_ocsserver:
505  ret, machine_uuid = result
506  AttemptToScheduler(content, machine_uuid)
507  else :
508  ret = result
509 
510  if not ret:
511  self.logger.error("no inventory created!")
512  return False
513 
514  except IOError, e:
515  self.logger.exception(e)
516  if hasattr(e, 'message') and e.message != '':
517  self.logger.error(e.mesage)
518  except Exception, e:
519  self.logger.exception(e)
520 
521  return True
