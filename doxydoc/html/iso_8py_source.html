204  def _cbSuccess(result):
205  out, err, code = result
206  if code:
207  self.logger.error('Error while generating ISO image: error code = %d' % code)
208  self.logger.error('stderr: %s' % err)
209  ret = False
210  else:

212  ret = True

215  def _cbError(result):
216  out, err, signalNum = result
217  self.logger.error('Signal %d received while generating ISO image' % signalNum)
218  return False
219 
220  for medianumber in self.medialist:
221  args = ['-f', '-v', '-v', '-R']
222  # Volume ID
223  args += ['-V', 'FIXME TITLE']
224  # Application ID

228  # Other options

233  args.append(
