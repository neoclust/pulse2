331  def parse(self, query):
332  p1 = re.compile('\|\|')
333  p2 = re.compile('::')
334  p3 = re.compile('==')
335  p4 = re.compile(', ')
336  p5 = '(^>.+<$)'
337  p_sep_plural = '(^>|<$)' # multiple entries are surounded by > and <
338 
339  queries = p1.split(query)
340  ret = []
341  for q in queries:
342  a = p2.split(q)
343  b = p3.split(a[0])
344  c = p3.split(a[1])
345  if re.search(p5, c[1]) != None:
346  c[1] = re.sub(p_sep_plural, '', c[1])
347  val = p4.split(c[1])
348  if len(val) == 1:
349  val = val[0]
350  ret.append([b[0], b[1], c[0], val])
351  return ret
