59  def initialize(self, xmlmapping):
60  self.doc = parse(xmlmapping)
61  self.tables = {}
62  self.nomenclatures = {}
63 
64  for table in self.doc.documentElement.getElementsByTagName("MappedObject"):
65  xmlname = table.getAttribute('name')
66  xmlclass = table.getAttribute('class')
67  if xmlclass == 'Null':
68  continue
69  self.tables[xmlname] = [xmlclass, {}]
70  for field in table.getElementsByTagName('MappedField'):
71  xmlfrom = field.getAttribute('from')
72  xmlto = field.getAttribute('to')
73  self.tables[xmlname][1][xmlfrom] = xmlto
74  if field.hasAttribute('type') and field.getAttribute('type') == 'nomenclature':
75  self.tables[xmlname][1][xmlfrom] = ('nom%s%s'%(xmlclass, xmlto), xmlto)
76  if not xmlclass in self.nomenclatures:
77  self.nomenclatures[xmlclass] = {}
78  self.nomenclatures[xmlclass][xmlto] = True

80  def parseSafe(self, xmlstring):
81  """
82  Try to parse the file with other encodings
83  """
84  xml = None
85  for encoding in ['ISO-8859-1', 'UTF-8']:
86  self.logger.info('Trying to parse with enforced %s encoding'
87  % encoding)
88  try:
89  builder = EncodingSafeParser()
90  builder.setEncoding(encoding)
91  xml = builder.parseString(xmlstring)
92  self.logger.info('String successfully parsed')
93  # Exit for loop if success
94  break
95  except expat.ExpatError, e:
96  self.logger.error('Parsing failed')
97  if expat.errors.XML_ERROR_INVALID_TOKEN in str(e):
98  pass
99  else:
100  # Unhandled error, just re-raise it
101  raise e
102  if xml:
103  return xml
104  else:
105  raise Exception("Can't parse inventory XML string")

107  def parse(self, xmltext):
108  """
109  Parse the given XML string which is the content of a OCS inventory.
110  """
111  self.logger = logging.getLogger()
112  xml = None
113  try:
114  xml = parseString(xmltext)
115  except expat.ExpatError, e:
116  self.logger.error("Can't parse inventory XML string")
117  if expat.errors.XML_ERROR_INVALID_TOKEN in str(e):
118  self.logger.error('The XML string may use a wrong encoding')
119  else:
120  # Unhandled error, just re-raise it
121  raise e
122  if not xml:
123  # Let's try another encoding
124  xml = self.parseSafe(xmltext)
125  inventory = {}
126  for tablename in self.tables:
127  in_network = tablename == u'NETWORKS'
128  try:
129  dbtablename = self.tables[tablename][0]
130  inventory[dbtablename] = []
131  #tag node list
132  for tag in xml.getElementsByTagName(tablename):
133  entry = {}
134  if in_network:
135  # Skip lo interface and network device with a 127.x.x.x
136  # address.
137  try:
138  netif = tag.getElementsByTagName('DESCRIPTION')[0].childNodes[0].nodeValue
139  except:
140  netif = ''
141  try:
142  ip = tag.getElementsByTagName('IPADDRESS')[0].childNodes[0].nodeValue
143  except:
144  ip = ''
145  if netif == 'lo' or ip.startswith('127.'):
146  self.logger.debug('Skipping computer local interface from inventory')
147  continue
148  for fieldname in self.tables[tablename][1]:
149  try:
150  field = tag.getElementsByTagName(fieldname)[0]
151  dbfieldname = self.tables[tablename][1][fieldname]
152  entry[dbfieldname] = field.childNodes[0].nodeValue
153  except IndexError:
154  pass
155  inventory[dbtablename].append(entry)
156  except IndexError:
157  pass
158  registerval={}
159  for EntryRegister in inventory['Registry']:
160  for EntryRegister1 in EntryRegister:
161  if EntryRegister1 != 'Value':
162  registerkey=EntryRegister[EntryRegister1]
163  else:
164  registerval[registerkey]=EntryRegister[EntryRegister1]
165  inventory[u'RegistryInfos']=[registerval]
166  return inventory
