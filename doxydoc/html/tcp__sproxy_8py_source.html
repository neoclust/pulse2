141  def get_all_interfaces():
142  """
143  Return list of all available interfaces with IP
144  @see: https://gist.github.com/pklaus/289646
145  """
146  def format_ip(addr):
147  return '.'.join([str(ord(x)) for x in addr])
148  max_possible = 128 # arbitrary. raise if needed.
149  bytes = max_possible * 32
150  s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
151  names = array.array('B', '\0' * bytes)
152  outbytes = struct.unpack('iL', fcntl.ioctl(
153  s.fileno(),
154  0x8912, # SIOCGIFCONF
155  struct.pack('iL', bytes, names.buffer_info()[0])
156  ))[0]
157  namestr = names.tostring()
158  lst = []
159  for i in range(0, outbytes, 40):
160  name = namestr[i:i+16].split('\0', 1)[0]
161  ip = format_ip(namestr[i+20:i+24])
162  if not ip.startswith('127.'):
163  lst.append({
164  'name': name,
165  'ip': ip,
166  })
167  return lst
