118 def read_logs(logfiles, start_date, stop_date):
119  """
120  Read scheduler logs in log_dir from start_date to stop_date
121  """
122 
123  loadavg = {}
124  fds = {}
125  memory = {}
126  db = {}
127 
128  # Parse all log files in the directory
129  for logfile in logfiles:
130 
131  try:
132  fh = open(logfile)
133  except IOError:
134  print "can't read %s " % logfile
135  continue
136 
137  print "parsing %s ... " % logfile
138  for line in fh: # Parse each line in the log file
139  # Add the "BALANCE" test to avoid computing regexp if the line doesn't match
140  res = re.search(HEALTH_REGEX, line)
141 
142  if not res: # give up if line do not match
143  continue
144 
145  stamp = int(time.mktime(time.strptime(res.group(1), '%Y-%m-%d %H:%M')))
146  if (stamp < start_date or stamp > stop_date): # give up if time do not match
147  continue
148 
149  scheduler = res.group(2)
150  dump = eval(res.group(3))
151 
152  if 'db' in dump:
153  if not scheduler in db:
154  db[scheduler] = {}
155  if not stamp in db[scheduler]:
156  db[scheduler][stamp] = {}
157  db[scheduler][stamp] = int(dump['db']['checkedinconns'])
158  else:
159  db[scheduler][stamp] += int(dump['db']['checkedinconns'])
160 
161  if 'memory' in dump:
162  if not scheduler in memory:
163  memory[scheduler] = {}
164  if not stamp in memory[scheduler]:
165  memory[scheduler][stamp] = {}
166  memory[scheduler][stamp] = int(dump['memory']['free'])
167  else:
168  memory[scheduler][stamp] += int(dump['memory']['free'])
169 
170  if 'loadavg' in dump:
171  if not scheduler in loadavg:
172  loadavg[scheduler] = {}
173  if not stamp in loadavg[scheduler]:
174  loadavg[scheduler][stamp] = int(dump['loadavg']['1min'])
175  else:
176  loadavg[scheduler][stamp] += int(dump['loadavg']['1min'])
177 
178  if 'fd' in dump:
179  if not scheduler in fds:
180  fds[scheduler] = {}
181  if not stamp in fds[scheduler]:
182  fds[scheduler][stamp] = sum(dump['fd'].values())
183  else:
184  fds[scheduler][stamp] += sum(dump['fd'].values())
185 
186  return (memory, loadavg, fds, db)
