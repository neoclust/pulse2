39 def get_my_network ():
40  """
41  Getting the networking info - directly from test machine

43  Another way to get necesary networking infomations to compare getted
44  results of tested methods.
45 
46  @return: IP address, netmask and default gateway of testmachine

58 def get_my_gateway():
59  import socket
60  import struct
61 
62  with open("/proc/net/route") as f_route:
63  for line in f_route:
64  fields = line.strip().split()
65  if fields[1] != '00000000' or not int(fields[3], 16) & 2:
66  continue
67  return socket.inet_ntoa(struct.pack("<L", int(fields[2], 16)))

70 def exec_cmd(cmd, exclude=None):
71  """
72  Command execute and output parse
73 
74  @param cmd: command line expression to execute
75  @type cmd: str
76 
77  @param exclude: expression to exclude from output
78  @type exclude: str
79 
80  @return: output from executed expression
81  @rtype: str
82 
83  """
84  ps = subprocess.Popen(cmd, shell=True, stdout=subprocess.PIPE)
85  out = ps.communicate()
86 
87  ret = []
88  for line in out :
89  if not line :
90  continue
91  line = line.replace("\n", "")
92  if exclude :
93  if exclude in line :
94  line = line.replace(exclude, "")
95 
96  ret.append(line)
97  return ret[0]
