31 class StatisticsProcessing :
32  """
33  Provides the periodical and final updates of global command statistics.
34 
35  Statistics are periodically updated to have a global sumarization
36  of states of commands on host.
37 
38  Final statistics update is called on the last switching a circuit
39  to overtimed state. This update is scheduled with a delay having
40  the possibility to cancel if the previous updated circuit wasn't
41  the last definitive circuit of updated command.
42  """
43  # global statistics container of valid commands
44  stats = {}
45  # scheduled final statistics updates
46  wdogs = {}
47 
48  # previous commands (missing command == expired)
49  previous = []

56  def update(self, cmd_id=None):

59  self.stats[cmd_id] = self._get_stats(cmd_id)[cmd_id]

65  def check_and_schedule_for_expired(self):
66  """Looks for expired commands and schedules an update"""
67  for cmd_id in self.previous :
68  if cmd_id not in self.stats :
69  # during previous awake presented, so now expired
70  self.watchdog_schedule(cmd_id)

76  def _get_stats(self, cmd_id=None):
77  """
78  Looking for statistics for selected or all valid commands.
79 
80  @param cmd_id: id of Commands record
81  @type cmd_id: int
82  """
83  __stats = get_commands_stats(self.config.name, cmd_id)
84 
85  stats = {}
86 
87  if cmd_id :
88  all_states = dict([(q[1], q[2]) for q in __stats if q[0]==cmd_id])
89  stats[cmd_id] = all_states
90  else:
91  cmd_ids = []
92  [cmd_ids.append(k[0]) for k in __stats]
93 
94  for cmd_id in cmd_ids :
95  all_states = dict([(q[1], q[2]) for q in __stats if q[0]==cmd_id])
96  stats[cmd_id] = all_states
97 
98  return stats

100  def _update_for(self, cmd_id):
101  """
102  Updates the global statistics of selected command
103 
104  @param cmd_id: id of Commands record
105  @type cmd_id: int
106  """
107  try:
108  process_non_valid(self.config.name,
109  self.config.non_fatal_steps)
110 
111  all_stats = self._get_stats(cmd_id)
112  stats = all_stats[cmd_id]
113 
114  update_commands_stats(cmd_id, stats)
115 
116 
117  del self.wdogs[cmd_id]
118  self.logger.info("Final statistics updated for command: %s" % cmd_id)
119 
120  except Exception, e:
121  self.logger.warn("Statistics update failed: %s" % str(e))

123  def watchdog_schedule(self, cmd_id):
124  """
125  Schedules the final stats update.
126 
127  @param cmd_id: id of Commands record
128  @type cmd_id: int
129  """
130  if cmd_id in self.wdogs :
131  call_id = self.wdogs[cmd_id]
132  call_id.cancel()
133  self.logger.debug("Statistics: schedule cancelled: %s" % cmd_id)
134  self.logger.debug("Statistics: scheduling the final update for command %s" % cmd_id)
135  call_id = reactor.callLater(10, self._update_for, cmd_id)
136 
137  self.wdogs[cmd_id] = call_id
