121 def read_logs(log_dir, start_date, stop_date):
122  """
123  Read launchers logs in log_dir from start_date to stop_date
124  """
125 
126  regexp = "([0-9]{4}-[0-9]{2}-[0-9]{2} [0-9]{2}:[0-9]{2}):[0-9]{2},.*launcher ([^:]*): BALANCE: (.*)"
127  groups = {}
128  launchers = {}
129 
130  # Parse all log files in the directory
131  for file in os.listdir(log_dir):
132 
133  # If this is a real log file
134  # TODO : better test
135  if ".log" in file:
136 
137  print "reading %s"%file
138  fh = open(os.path.join(log_dir,file))
139 
140  # Parse each line in the log file
141  for line in fh:
142 
143  # Add the "BALANCE" test to avoid computing regexp if the line doesn't match
144  if (("BALANCE:" in line) and re.search(regexp, line)) :
145 
146  # Extract regexp results
147  res = re.search(regexp, line)
148  date = res.group(1)
149  stamp = int(time.mktime(time.strptime(date, '%Y-%m-%d %H:%M')))
150 
151  if (stamp > start_date and stamp < stop_date):
152 
153  launcher = res.group(2)
154  stats = res.group(3)
155  # Convert str to dict
156  dump = eval(stats)
157 
158  if 'global' in dump:
159 
160  if not launcher in launchers:
161  launchers[launcher] = {}
162  if stamp in launchers[launcher]:
163  launchers[launcher][stamp] += int(dump['global']['running'])
164  else:
165  launchers[launcher][stamp] = int(dump['global']['running'])
166 
167  if 'by_group' in dump:
168 
169  # For each group available
170  for group in dump['by_group'].keys():
171  if not group in groups:
172  groups[group] = {}
173  # If this group as already an entry for this time (from an other launcher), we add the new one
174  if stamp in groups[group]:
175  groups[group][stamp] += int(dump['by_group'][group]['running'])
176  else:
177  groups[group][stamp] = int(dump['by_group'][group]['running'])
178  return (groups,launchers)
