48 def getPubKey(key_name):
49  """
50  Handle remote download of this launcher's pubkey.
51  key_name is as define in the config file
52  """
53  try:
54  LauncherConfig().ssh_keys[key_name]
55  except KeyError:
56  key_name = LauncherConfig().ssh_defaultkey
57 
58  if key_name == None or key_name == '':
59  key_name = LauncherConfig().ssh_defaultkey
60  try:
61  ssh_key = open(LauncherConfig().ssh_keys[key_name] + '.pub')
62  except IOError: # key does not exists, give up
63  return ''
64  ret = ' '.join(ssh_key)
65  ssh_key.close()
66  return ret

68 def setDefaultClientOptions(client):
69  """
70  client is a simple dict, which should contain required connexion infos, for now:
71  group: an optional group membership
72  server_check: an dict of stuff-to-check-on-client (see pulse2-output-wrapper)
73  client_check: an dict of stuff-to-check-on-client (see pulse2-output-wrapper)
74  protocol: which one to use for connexion, mandatory
75  host: where to connect, mandatory
76  port: default depends on chosen protocol
77  user: when auth is needed, default is root
78  passwd: when auth is needed, default is "" (empty string)
79  cert: when auth is needed, default depends on chosen protocol
80  options: array of strings to pass to the connexion initiator
81  rootpath: used to know where to perform operations ('/' under Unix,
82  '/cygdrive/c' under MS/Win, etc ...
83  """
84  # FIXME: handle missing keys
85 
86  # client group, used to define 'targets' groups to gather aggregated stats
87  if not 'group' in client:
88  client['group'] = None
89 
90  if not 'server_check' in client:
91  client['server_check'] = None
92 
93  if not 'client_check' in client:
94  client['client_check'] = None
95 
96  if not 'action' in client:
97  client['action'] = None
98 
99  # command execution throught SSH
100  if client['protocol'] == 'ssh':
101  if not 'port' in client:
102  client['port'] = 22
103  if not 'user' in client:
104  client['user'] = 'root'
105  if not 'cert' in client:
106  client['cert'] = LauncherConfig().ssh_keys[LauncherConfig().ssh_defaultkey]
107  client['transp_args'] = ['-T', '-o', 'IdentityFile=%s' % client['cert'], '-o', 'User=%s' % client['user']]
108  for option in LauncherConfig().ssh_options:
109  client['transp_args'] += ['-o', option]
110  if LauncherConfig().ssh_forward_key == 'always' or \
111  LauncherConfig().ssh_forward_key == 'let' and 'forward_key' in client:
112  client['transp_args'] += ['-A']
113  else:
114  client['transp_args'] += ['-a']
115 
116  # TCP forwarding through SSH, mainly used for VNC proxying
117  if client['protocol'] == 'tcpsproxy':
118  if not 'port' in client:
119  client['port'] = 22
120  if not 'user' in client:
121  client['user'] = 'root'
122  if not 'cert' in client:
123  client['cert'] = LauncherConfig().ssh_keys[LauncherConfig().ssh_defaultkey]
124  client['transp_args'] = ['IdentityFile=%s' % client['cert'], 'User=%s' % client['user']]
125  for option in LauncherConfig().ssh_options:
126  client['transp_args'] += [option]
127  client['transp_args'] += ["UserKnownHostsFile=/dev/null", 'StrictHostKeyChecking=no'] # required to prevent tcp forwarding failure
128 
129  # Push/pull mode using wget driven by SSH
130  if client['protocol'] == 'wget':
131  if not 'port' in client:
132  client['port'] = 22
133  if not 'user' in client:
134  client['user'] = 'root'
135  if not 'cert' in client:
136  client['cert'] = LauncherConfig().ssh_keys[LauncherConfig().ssh_defaultkey]
137  if not 'proto_args' in client:
138  client['proto_args'] = ['-nv']
139  if not LauncherConfig().wget_check_certs:
140  client['proto_args'] += ['--no-check-certificate']
141  if LauncherConfig().wget_resume:
142  client['proto_args'] += ['-c']
143  if LauncherConfig().wget_options != '':
144  client['proto_args'] += LauncherConfig().wget_options
145  if 'maxbw' in client: # FIXME: handle low values of BWLimit (see mechanism below for rsync)
146  client['proto_args'] += ['--limit-rate', '%d' % int(client['maxbw'] / 8) ] # bwlimit arg in B/s
147  client['transp_args'] = ['-T', '-o', 'IdentityFile=%s' % client['cert'], '-o', 'User=%s' % client['user']]
148  for option in LauncherConfig().ssh_options:
149  client['transp_args'] += ['-o', option]
150 
151  # Local Proxy mode (obviously using rsync)
152  if client['protocol'] == 'rsyncproxy':
153  if not 'port' in client:
154  client['port'] = 22
155  if not 'user' in client:
156  client['user'] = 'root'
157  if not 'cert' in client:
158  client['cert'] = LauncherConfig().ssh_keys[LauncherConfig().ssh_defaultkey]
159  client['transp_args'] = ['-T', '-o', 'IdentityFile=%s' % client['cert'], '-o', 'User=%s' % client['user']]
160  for option in LauncherConfig().ssh_options:
161  client['transp_args'] += ['-o', option]
162  client['transp_args'] += ['-A'] # always forward TCP key
163  client['transp_args'] += ['-o', 'UserKnownHostsFile=/dev/null', '-o', 'StrictHostKeyChecking=no'] # required to prevent key forwarding failure
164  if not 'proto_args' in client:
165  client['proto_args'] = ['--archive', '--verbose']
166  # inside ssh get the same args as outside ssh
167  client['proto_args'] += ['--rsh="%s %s"' % (LauncherConfig().ssh_path, ' '.join(map(lambda x:'-o %s' % x, LauncherConfig().ssh_options)))]
168 
169  if client['protocol'] == 'rsyncssh':
170  if not 'port' in client:
171  client['port'] = 22
172  if not 'user' in client:
173  client['user'] = 'root'
174  if not 'cert' in client:
175  client['cert'] = LauncherConfig().ssh_keys[LauncherConfig().ssh_defaultkey]
176  client['transp_args'] = ['-o', 'IdentityFile=%s' % client['cert'], '-o', 'User=%s' % client['user']]
177  if not 'proto_args' in client:
178  client['proto_args'] = ['--archive', '--verbose']
179  if not LauncherConfig().is_rsync_limited:
180  client['proto_args'].extend(['--no-group', '--no-owner', '--chmod=%s'%(LauncherConfig().rsync_set_chmod)])
181  if LauncherConfig().rsync_resume:
182  client['proto_args'] += ['--partial', '--append']
183  for option in LauncherConfig().ssh_options:
184  client['transp_args'] += ['-o', option]
185  client['proto_args'] += ['--rsh', ' '.join([LauncherConfig().ssh_path] + client['transp_args'])]
186  if 'maxbw' in client:
187  if client['maxbw'] == 0: # bwlimit forced to 0 => no BW limit
188  pass
189  else:
190  bwlimit = int(client['maxbw'] / (1024 * 8))
191  if bwlimit < 1:
192  bwlimit = 1 # as bwlimit = 0 imply no limit, min bwlimit set to 1
193  client['proto_args'] += ['--bwlimit', '%d' % bwlimit] # bwlimit arg in kB/s
194  return client
