30 class LoopingStarter(object):
31  """
32  Dispatching the batchs of start.
33 
34  Each circuit must be started in a separated thread and each launching
35  must be delayed to avoid the saturation of launcher(s).
36  """
37 
38  # looping call reference
39  loop = None

41  def __init__(self, dispatcher, emitting_period):
42  """
43  @param dispatcher: main dispatcher reference
44  @type dispatcher: MscDispatcher
45 
46  @param emitting_period: delay inserted among the starts
47  @type emitting_period: float
48  """
49  self.dispatcher = dispatcher
50  self.emitting_period = emitting_period
51  self.logger = logging.getLogger()

54  def _run_one(self, circuit):
55  """
56  Executes the circuit in the thread.
57 
58  @param circuit: circuit to run
59  @type circuit: Circuit
60 
61  @return: start result
62  @rtype: Deferred
63  """
64 
65  d = deferToThread(circuit.run)
66  @d.addErrback
67  def eb(reason):
68  """ Thread start fallback """
69  self.logger.error("Circuit #%s: start failed: %s" % (circuit.id, reason))
70 
71  @d.addCallback
72  def cb(reason):
73  """ Thread start callback """
74  self.dispatcher._circuits.append(circuit)
75 
76  return d

79  def _run_later(self, circuits):
80  """
81  Starts the each circuit with a predefined delay.
82 
83  This periodic call avoids a saturation on the server (i.e. launcher).
84  @param circuits: circuits to start
85  @type circuits: iterator
86  """
87  try :
88  circuit = next(circuits)
89  self._run_one(circuit)
90 
91  except StopIteration :
92  if self.loop.running:
93  self.loop.stop()
94 
95  self.logger.debug("circuits started")
96

98  def run(self, circuits):
99  """
100  Executes all the circuits.
101 
102  @param circuits: circuits to run
103  @type circuits: list
104 
105  @return: list of start results
106  @rtype: list
107  """
108  try :
109  if len(circuits) == 0:
110  self.logger.info("Nothing to execute")
111  return True
112 
113  self.loop = LoopingCall(self._run_later, iter(circuits))
114  d = self.loop.start(self.emitting_period)
115  d.addErrback(self._loop_fail)
116 
117  return d
118 
119  except Exception, e :
120  self.logger.error("Circuits start failed: %s" % str(e))
121  return False

125  self.logger.error("Loop call starting failed: %s" % str(failure))
