174 def getAllSoftwares(ctx, softname='', vendor=None):
175  def replace_splat(param):
176  if '*' in param:
177  return param.replace('*', '%')
178  return param
179 
180  def check_param(param):
181  if param == '' or param == '*' or param == '%':
182  return None
183  return replace_splat(param)
184 
185  software = check_param(softname)
186  if vendor is not None:
187  vendor = check_param(vendor)
188  if software is None:
189  software = '%'
190  return [x[0] for x in Glpi().getAllSoftwares(ctx,
191  softname=softname,
192  vendor=vendor,
193  limit=20)]
