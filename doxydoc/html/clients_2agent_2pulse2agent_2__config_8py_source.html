58  def getlist(self, section, option):
59  """
60  Transforms string containing a list to true list datatype.
61 
62  This method has a same behavior as others like, getint, getbool, etc.
63  The content of declared list may contain several base datatypes like
64  str, bool, int, float. When detection fails, element will be converted
65  to str.
66 
67  @param section: name of section
68  @type section: str
69 
70  @param option: name of option
71  @type option: str
72 
73  @return: list of detected values
74  @rtype: list
75  """
76  raw = self.get(section, option)
77  ret = []
78 
79  # try to check and convert type of all elements
80  for s in raw.split(","):
81  element = s.strip()
82  if element.lower() == "true":
83  ret.append(True)
84  continue
85  elif element.lower() == "false":
86  ret.append(False)
87  continue
88 
89  float_value = None
90  try:
91  float_value = float(element)
92  except ValueError:
93  # float conversion failed - element typed as str
94  ret.append(element)
95  continue
96  try:
97  int_value = int(element)
98  except ValueError:
99  if float_value is None:
100  ret.append(element)
101  continue
102 
103  # because int value can be converted to float too,
104  # comapring of both values resolves a final datatype
105  # i.e.:
106  # int("754") == float("754")
107  # int("124.251") != float("124.251")
108  if int_value == float_value:
109  ret.append(int_value)
110  else:
111  ret.append(float_value)
112 
113  return ret

117 class ConfigReader(type):
118  """
119  Provides a redable declaration of defaults for ConfigParser instances.
120 
121  The style of declarations are realized as a root class
122  containing nested classes which will be detected as option containers.
123  Name of nested class is considered as name of section and its attributtes
124  defines the options as defaults.
125  Method read() loads a config file with ConfigParser syntax and overrides
126  the values of options.
127 
128  All options will be finally accessible hierarchically, like this:
129 
130  >>> config.section.option
131 
132  Some rules for a final declaration:
133  - Root frame of must declare this frame as metaclass
134  - Each section must be declared as nested class of root
135  - Nested class inherits from object
136  - Assigning of attributte defines its datatype
137 
138  Example :
139  ---------
140  >>> class Config(object):
141  >>> __metaclass__ = ConfigReader
142  >>>
143  >>> class database(type):
144  >>> host = "localhost"
145  >>> port = 3306
146  >>>
147  >>> config = Config()
148  >>> config.database.port
149  3306
150  >>>
151  """
152

174  def read(self, filename):
175  """
176  Reads the content of config file and overrides the defaults.
177 
178  @param filename: path to config file
179  @type filename: str
180  """
181 
182  config_file = ExtendedConfigParser()
183  config_file.read(filename)
184 
185  for section_name in config_file.sections():
186  if hasattr(self, section_name):
187  section = getattr(self, section_name)
188  if inspect.isclass(section):
189 
190  setattr(section, "__name__", section_name)
191  self._update_options(config_file, section)
192 
193  else:
194  raise InvalidSection(section_name)
195  else:
