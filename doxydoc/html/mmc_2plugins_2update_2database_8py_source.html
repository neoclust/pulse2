165  @DatabaseHelper._listinfo
166  @DatabaseHelper._session
167  def get_updates(self, session, params):
168  """
169  Get all updates by the specified filters
170  Status filter is specially treaten
171  in this order of priority: Update -> Update Type
172  """
173  #.add_column(func.count(Target))\
174  try:
175  # Defining subqueries
176  installed_targets = session.query(Target.update_id, func.sum(Target.is_installed).label('total_installed')).group_by(Target.update_id).subquery()
177  all_targets = session.query(Target.update_id, func.count('*').label('total_targets')).group_by(Target.update_id).subquery()
178 
179 
180  # Main query
181  query = session.query(Update, func.ifnull(installed_targets.c.total_installed, 0).label('total_installed'), func.ifnull(all_targets.c.total_targets, 0).label('total_targets'))\
182  .join(Update.update_type)\
183  .outerjoin(installed_targets, Update.id == installed_targets.c.update_id)\
184  .outerjoin(all_targets, Update.id == all_targets.c.update_id)
185 
186  # ==== STATUS FILTERING ======================
187  if 'filters' in params and 'status' in params['filters']:
188  # Special filter treatment for status
189  Status = params['filters']['status']
190  del params['filters']['status']
191 
192  # other non-confusing filters
193  # are automatically treaten by @DatabaseHelper._listinfo
194 
195  if Status == STATUS_NEUTRAL:
196  # Neutral status
197  query = query.filter(\
198  (Update.status == Status) & \
199  (UpdateType.status == Status)\
200  )
201  else:
202  # Dominant status
203  query = query.filter(\
204  (Update.status == Status)| \
205  (\
206  (Update.status == STATUS_NEUTRAL) &\
207  (UpdateType.status == Status)\
208  )\
209  )
210  #
211  # ==== END STATUS FILTERING ==================
212 
213  return query
214  except Exception, e:
215  logger.error("DB Error: %s" % str(e))
216  return False

239  @DatabaseHelper._session
240  def get_updates_for_host_by_dominant_status(self, session, uuid, dStatus, is_installed=None):
241  """
242  Get all update to install for host,
243  this function return all updates for the host
244  if the update have the Status (STATUS_ENABLED or STATUS_DISABLED, STATUS_NEUTRAL)
245  in this order of priority:
246  Target -> Update -> Update Type
247  """
248 
249  if dStatus == STATUS_NEUTRAL:
250  logger.error("Neutral status is not accepted, use get_neutral_updates function instead")
251  return False
252 
253  try:
254  query = session.query(Target)\
255  .add_entity(Update).join(Update)\
256  .join(UpdateType)\
257  .filter(Target.uuid == uuid)
258 
259  if is_installed is not None:
260  query = query.filter(Target.is_installed == is_installed)
261 
262  # ============================================
263  # ==== STATUS FILTERING ======================
264  # ============================================
265  query = query.filter(\
266  # 1st level filtering : Target status
267  (Target.status == dStatus) |\
268  (\
269  (Target.status == STATUS_NEUTRAL) &\
270  (\
271  # 2nd level filtering : Update status
272  (Update.status == dStatus)| \
273  (\
274  (Update.status == STATUS_NEUTRAL) &\
275  (UpdateType.status == dStatus)\
276  )\
277  )\
278  )
279  )
280  # ============================================
281  # ==== END STATUS FILTERING ==================
282  # ============================================
283 
284  result = []
285 
286  for (target, update) in query:
287  result.append(update.toDict())
288  return result
289  except Exception, e:
290  logger.error("DB Error: %s" % str(e))
