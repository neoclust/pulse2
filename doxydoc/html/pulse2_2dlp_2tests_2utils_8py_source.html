62 class HTTPClient(object):
63 
64  def __init__(self, base_url):
65  self.cookie_jar = cookielib.CookieJar()
66  self.opener = urllib2.build_opener(urllib2.HTTPCookieProcessor(self.cookie_jar))
67  urllib2.install_opener(self.opener)
68  self.base_url = base_url
69 
70  def get(self, url, headers={}):
71  """HTTP GET
72 
73  url should be a string containing a valid URL.
74  headers should be a dictionary
75  """
76  url = self.base_url + url
77  request = urllib2.Request(url, headers=headers)
78  return self.execute_request(request)
79 
80  def post(self, url, data=None, headers={}):
81  """HTTP POST
82 
83  url should be a string containing a valid URL.
84  data should be a url-encodable dictionary
85  headers should be a dictionary
86  """
87  url = self.base_url + url
88  if data is None:
89  postdata = None
90  else:
91  postdata = urllib.urlencode(data, True)
92  request = urllib2.Request(url, postdata, headers)
93  return self.execute_request(request)
94 
95  def execute_request(self, request):
96  response = self.opener.open(request)
97  response.status_code = response.getcode()
98  response.data = response.read()
99  if response.info().get("Content-Type", None) == "application/json":
100  response.data = json.loads(response.data)
101  return response
