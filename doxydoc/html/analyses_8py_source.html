251  def local_proxy_upload_status(self, cohq):
252  """
253  attempt to analyse coh in the same command in order to now how we may advance.
254  possible return values:
255  - 'waiting': my time is not yet come
256  - 'server': I'm an active proxy server
257  - 'dead': I'm a client and all proxies seems dead
258  - 'error': Something wrong was found in the command (usually mess in priorities)
259  - an int: I'm a client and the returned value is the CoH I will use
260  """
261  cmd_id = cohq.cmd.id
262  proxy_mode = self.get_proxy_mode_for_command(cmd_id)
263  if proxy_mode == 'queue':
264  return self.local_proxy_attempt_queue_mode(cohq)
265  elif proxy_mode == 'split':
266  return self.local_proxy_attempt_split_mode(cohq)
267  else:
268  self.logger.debug("scheduler %s: command #%s seems to be wrong (bad priorities?)" % (self.config.name, cohq.cmd.id))
269  return 'dead'

272  def get_proxy_mode_for_command(self, cmd_id):
273  """
274  Preliminar iteration to gather information about this command
275  the idea being to obtain some informations about what's going on
276  we are looking for the following elements
277  - the amount of priorities:
278  + only one => split mode (returns "split")
279  + as many as proxies => queue mode (returns "queue")
280  + no / not enough priorities => error condition (returns False)
281  """
282 
283  spotted_priorities = {}
284 
285  circuits = [c for c in self.get_active_circuits([cmd_id]) if c.qm.cmd.id==cmd_id]
286  for c in circuits :
287  if c.qm.coh.order_in_proxy != None: # some potential proxy
288  if c.qm.coh.order_in_proxy in spotted_priorities:
289  spotted_priorities[c.qm.coh.order_in_proxy] += 1
290  else:
291  spotted_priorities[c.qm.coh.order_in_proxy] = 1
292 
293  if len(spotted_priorities) == 0:
294  return False
295  elif len(spotted_priorities) == 1: # only one priority for all => split mode
296  self.logger.debug("scheduler %s: command #%s is in split proxy mode"
297  % (self.config.name, cmd_id))
298  return 'split'
299  elif len(spotted_priorities) == reduce(lambda x, y: x+y, spotted_priorities.values()): # one priority per proxy => queue mode
300  self.logger.debug("scheduler %s: command #%s is in queue proxy mode"
301  % (self.config.name, cmd_id))
302  return 'queue'
303  else: # other combinations are errors
304  self.logger.debug("scheduler %s: can'f guess proxy mode for command #%s" % (self.config.name, cmd_id))
305  return False

308  """split mode (parallel) implementation of proxy mode"""
309 
310  def __processProbes(result):
311  # remove bad proxy (result => alive_proxy):
312  alive_proxies = dict()
313  for (success, (probe, uuid, coh_id)) in result:
314  if success: # XMLRPC call do succeedeed
315  if probe:
316  if probe != "Not available":
317  alive_proxies[uuid] = coh_id
318 
319  # map if to go from {uuid1: (coh1, max1), uuid2: (coh2, max2)} to ((uuid1, max1), (uuid2, max2))
320  # ret val is an uuid
321  final_uuid = LocalProxiesUsageTracking().take_one(alive_proxies.keys(), cohq.cmd.id)
322  if not final_uuid: # not free proxy, wait
323  self.logger.debug("scheduler %s: coh #%s wait for a local proxy for to be usable" % (self.config.name, cohq.coh.id))
324  return 'waiting'
325  else: # take a proxy in alive proxies
326  final_proxy = alive_proxies[final_uuid]
327  self.logger.debug("scheduler %s: coh #%s found coh #%s as local proxy, taking one slot (%d left)" % (self.config.name, cohq.coh.id, final_proxy, LocalProxiesUsageTracking().how_much_left_for(final_uuid, cohq.cmd.getId())))
328  return final_proxy

335  if cohq.coh.getOrderInProxy() is None: # I'm a client: I MUST use a proxy server ...
336  temp_dysfunc_proxy = list() # proxies with no data (UPLOADED != DONE)
337  def_dysfunc_proxy = list() # proxies with no data (UPLOADED != DONE) and which definitely wont't process further (current_state != scheduled)
338  available_proxy = list() # proxies with complete data (UPLOADED = DONE)

349  for c in circuits :
350  # got 4 categories here:
351  # - DONE and not DONE
352  # - scheduled and not scheduled
353  # => upload DONE, (scheduled or not): proxy free to use (depnding on nb of clients, see below)
354  # => upload !DONE + (failed, over_timed) => will never be available => defin. failed
355  # => upload !DONE + ! (failed or over_timed) => may be available in some time => temp. failed
356  if (c.qm.coh.isStateFailed() or c.qm.coh.isStateOverTimed() or \
357  c.qm.coh.isStateStopped()) :
358  def_dysfunc_proxy.append(c.qm.coh.id)
359  elif c.qm.phase.is_done():
360  available_proxy.append(c.qm.coh.id)
361  else:
362  temp_dysfunc_proxy.append(c.qm.coh.id)
363

365  if len(temp_dysfunc_proxy) == 0: # and others seems dead
366  self.logger.debug("scheduler %s: coh #%s won't likely be able to use a local proxy" % (self.config.name, cohq.coh.id))
367  return 'dead'
368  else:
369  self.logger.debug("scheduler %s: coh #%s wait for a local proxy to be ready" % (self.config.name, cohq.coh.id))
370  return 'waiting'
371

373

375  #proxyCoH = session.query(CommandsOnHost).get(proxy)
376  proxy_target = [c.qm.target for c in self.get_active_circuits([cmd_id]) if c.qm.coh.id==proxy]

390  deffered_list.append(d)
391  dl = DeferredList(deffered_list)
392  dl.addCallback(__processProbes)

395  else: # I'm a server: let's upload
396  self.logger.error("scheduler %s: coh #%s become local proxy server" % (self.config.name, cohq.coh.id))
397  return 'server'

401  """ queue mode (serial) implementation of proxy mode """

403  smallest_done_upload_order_in_proxy = None
404  best_ready_proxy_server_coh = None
405  potential_proxy_server_coh = None
406 
407  # iterate over CoH which
408  # are linked to the same command
409  # are not our CoH

411  circuits = [c for c in self.get_active_circuits([cmd_id])

413  and c.qm.coh.id != cohq.coh.id]
414 
415  for c in circuits :
416 
417 
418 
419  if c.qm.phase_is_done(): # got a pal which succeeded in doing its upload
420  if not c.qm.coh.order_in_proxy is None :# got a potent proxy server
421  if smallest_done_upload_order_in_proxy < c.qm.coh.order_in_proxy:
422  # keep its id as it seems to be the best server ever
423  smallest_done_upload_order_in_proxy = c.qm.coh.order_in_proxy
424  best_ready_proxy_server_coh = c.qm.coh.id
425 
426  elif not c.qm.coh.isStateFailed(): # got a pal which may still do something
427  if c.qm.coh.order_in_proxy != None: # got a potential proxy server
428  if cohq.coh.order_in_proxy is None: # i may use this server, as I'm not server myself
429  potential_proxy_server_coh = c.qm.coh.id
430  elif cohq.coh.order_in_proxy > c.qm.coh.order_in_proxy:
431  # i may use this server, as it has a lower priority than me
432  potential_proxy_server_coh = c.qm.coh.id
433 
434  # we now know:
435  # a proxy that may be used
436  # a proxy that might be used
437  # let's take a decision about our future
438 
439  if cohq.coh.getOrderInProxy() == None: # I'm a client: I MUST use a proxy server ...
440  if best_ready_proxy_server_coh != None: # ... and a proxy seems ready => PROXY CLIENT MODE
441  (current_client_number, max_client_number) = self.get_client_usage_for_proxy(best_ready_proxy_server_coh)
442  if current_client_number < max_client_number:
443  self.logger.debug("scheduler %s: found coh #%s as local proxy for #%s" %
444  (self.config.name, best_ready_proxy_server_coh, cohq.coh.id))
445  return best_ready_proxy_server_coh
446  else:
447  self.logger.debug("scheduler %s: found coh #%s as local proxy for #%s, but proxy is full (%d clients), so I'm waiting" %
448  (self.config.name, best_ready_proxy_server_coh, cohq.coh.id, current_client_number))
449  return 'waiting'
450  elif potential_proxy_server_coh != None: # ... and one may become ready => WAITING
451  self.logger.debug("scheduler %s: coh #%s still waiting for a local proxy to use" %
452  (self.config.name, cohq.coh.id))
453  return 'waiting'
454  else: # ... but all seems dead => ERROR
455  self.logger.debug("scheduler %s: coh #%s won't likely be able to use a local proxy" %
456  (self.config.name, cohq.coh.id))
457  return 'dead'
458  else: # I'm a server: I MAY use a proxy ...
459  if best_ready_proxy_server_coh != None: # ... and a proxy seems ready => PROXY CLIENT MODE
460  (current_client_number, max_client_number) = self.get_client_usage_for_proxy(best_ready_proxy_server_coh)
461  if current_client_number < max_client_number:
462  self.logger.debug("scheduler %s: found coh #%s as local proxy for #%s" %
463  (self.config.name, best_ready_proxy_server_coh, cohq.coh.id))
464  return best_ready_proxy_server_coh
465  else:
466  self.logger.debug("scheduler %s: found coh #%s as local proxy for #%s, but proxy is full (%d clients), so I'm waiting" %
467  (self.config.name, best_ready_proxy_server_coh, cohq.coh.id, current_client_number))
468  return 'waiting'
469  elif potential_proxy_server_coh: # ... but a better candidate may become ready => WAITING
470  self.logger.debug("scheduler %s: coh #%s still waiting to know if is is local proxy client or server" %
471  (self.config.name, cohq.coh.id))
472  return 'waiting'
473  else: # ... and other best candidates seems dead => PROXY SERVER MODE
474  self.logger.debug("scheduler %s: coh #%s become local proxy server" %
475  (self.config.name, cohq.coh.id))
476  return 'server'

500  def local_proxy_may_continue(self, cohq):
501  """ attempt to analyse coh in the same command in order to now how we may advance.
502 
503  Clean algorithm:
504  client => always cleanup
505  server => cleanup only if *everybody" are in one of the following state:
506  - upload done
507  - upload ignored
508  - failed
509  - over_timed
510  to prevent race condition, not check is perform to count only our clients but everybody client
511  """
512 
513  if cohq.coh.isLocalProxy(): # proxy server, way for clients to be done
514  self.logger.debug("scheduler %s: checking if we may continue coh #%s" % (self.config.name, cohq.coh.id))
515  our_client_count = 0
516  cmd_id = cohq.cmd.id
517  clients = [c for c in self.get_active_circuits([cmd_id])
518  if c.qm.cmd.id==cohq.cmd.id
519  and c.qm.coh.id != cohq.coh.id
520  and not c.qm.phase.is_done()
521  and not c.qm.coh.isStateDone()
522  and not c.qm.coh.isStateOverTimed()]
523 
524  if cohq.cmd.hasToUseQueueProxy():
525  our_client_count = len(clients)
526  self.logger.debug("scheduler %s: found %s coh to be uploaded in command #%s" %
527  (self.config.name, our_client_count, cohq.cmd.id))
528 
529  elif cohq.cmd.hasToUseSplitProxy():
530 
531  clients = [c for c in clients if c.qm.coh.order_in_proxy is None]
532  our_client_count = len(clients)
533 
534  self.logger.debug("scheduler %s: found %s coh to be uploaded in command #%s" %
535  (self.config.name, our_client_count, cohq.cmd.id))
536 
537 
538  # proxy tracking update
539  if our_client_count != 0:
540  LocalProxiesUsageTracking().create_proxy(cohq.target.getUUID(),
541  cohq.coh.getMaxClientsPerProxy(),
542  cohq.cmd.id)
543  self.logger.debug("scheduler %s: (re-)adding %s (#%s) to proxy pool" %
544  (self.config.name, cohq.target.getUUID(), cohq.cmd.getId()))
545  else:
546  LocalProxiesUsageTracking().delete_proxy(cohq.target.getUUID(),
547  cohq.cmd.getId())
548  self.logger.debug("scheduler %s: (re-)removing %s (#%s) from proxy pool" %
549  (self.config.name, cohq.target.getUUID(), cohq.cmd.getId()))
550  return our_client_count == 0
551  else:
552  return True
