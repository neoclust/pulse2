31 class GlpiProvisionerConfig(ProvisionerConfig):
32 
33  def readConf(self):
34  PROFILEACL = "profile_acl_"
35  ProvisionerConfig.readConf(self)
36  try:
37  self.doauth = self.getboolean(self.section, "doauth")
38  except:
39  pass
40  for option in self.options(self.section):
41  if option.startswith(PROFILEACL):
42  value = self.get(self.section, option)
43  if isfile(value):
44  acls = open(value, 'r').read().split('\n')
45  # Clean empty lines, and join them by :
46  value = ':' + (':'.join([x for x in acls if x.strip() and x[0] != '#']))
47  else:
48  self.profilesAcl[option.replace(PROFILEACL, "")] = value
49  self.profilesOrder = self.get(self.section, "profiles_order").split()
50 
51  def setDefault(self):
52  ProvisionerConfig.setDefault(self)
53  self.doauth = True
54  self.profilesAcl = {}
55  self.profilesOrder = []

72  def _cbProvisioning(self, auth, authtoken):
73  """
74  Provision the MMC user account with ACLs
75  """
76  if not auth:
77  self.logger.warning("User authentication with GLPI web interface failed, but going on with provisioning")
78  profiles = Glpi().getUserProfiles(authtoken.getLogin())
79  self.logger.debug("User '%s' GLPI's profiles: %s" % (authtoken.getLogin(), str(profiles)))
80  self.logger.debug("Profiles order (from ini configuration): %s" % (self.config.profilesOrder))
81  selected = None
82  for profile in self.config.profilesOrder:
83  if profile in profiles:
84  selected = profile
85  break
86  if not selected:
87  self.logger.info("User GLPI's profile can't be applied")
88  else:
89  self.logger.debug("Selected GLPI profile is %s" % selected)
90  try:
91  acls = self.config.profilesAcl[selected.lower()]
92  except KeyError:
93  acls = None
94  if not acls:
95  self.logger.info("No ACL to apply for the GLPI profile %s" % selected)
96  else:
97  l = ldapUserGroupControl()
98  self.logger.info("Setting MMC ACL corresponding to GLPI profile %s: %s" % (selected, acls))
99  uid = authtoken.getLogin()
100  entry = l.getDetailedUser(uid)
101  if not "lmcUserObject" in entry["objectClass"]:
102  entry["objectClass"].append("lmcUserObject")
103  l.changeUserAttributes(uid, "objectClass", entry["objectClass"])
104  l.changeUserAttributes(authtoken.getLogin(), "lmcAcl", acls)
105  return authtoken
