51 class RequestExtractor(object):
52  """
53  Extracts a request received from client.
54 
55  Description of format of request:
56  Received as tuple containing a global method identifier and its parameters,
57  method extract() returns a ready
58  """
59  DELIMITER = "."
60  args_types_allowed = [str, tuple, list, dict]
61 
62 
63  def extract(self, _request):
64  request = eval(_request)
65  if isinstance(request, list) or isinstance(request, tuple):
66  if len(request) == 2:
67  t_method, args = request
68  prefix, method = self._extract_method(t_method)
69  args = self._validate_args(args)
70  return prefix, method, args
71  else:
72  raise IndexError("Expected number of elements in request: 2")
73  else:
74  #print request, type(request)
75  raise TypeError("Invalid request format")
76 
77 
78  def _extract_method(self, text):
79  if isinstance(text, str):
80  if self.DELIMITER in text:
81  # allowed only one delimiter
82  # i.e.: pkgs.get_package
83  if text.count(self.DELIMITER) == 1:
84  # returns a prefix of endpoint and a method identifier
85  return text.split(self.DELIMITER)
86  else:
87  raise IndexError("Not allowed more than one method delimiter %s" % text)
88  else:
89  raise MethodWithoutPrefix(text)
90  else:
91  raise TypeError("Invalid format of method identifier, <str> type expected")
92 
93  def _validate_args(self, args):
94  if isinstance(args, str):
95  return [args]
96  for args_type in self.args_types_allowed:
97  if isinstance(args, args_type):
98  return args
99  else:
100  raise TypeError("Invalid format of args; one of <%s> expected" % repr(self.args_types_allowed))

163  Common point providing the cooperation between gateway and endpoints.
164 
165  Gateway as producer puts the received requests into a queue and launches
166  the getters of endpoints (consumers) by linked trigger.

169  a result is always returned to producer to pass them to client.
170  """

174  ]
