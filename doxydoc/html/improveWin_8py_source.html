57 '''

59 @param path: the path of the file
60 @param nIcon: position of the icon in the file
61 @param exeOrDll=True: set False if is a icon file
62 @return string bit of icon in Jpeg Format encoded in base64, else Error : None

64 def windowsIconGetter(path, nIcon = 0 , exeOrDll = True): #It's can be exe, dll, or already ico
65 
66  if os.path.isfile(path):
67  dst = cStringIO.StringIO()
68  if exeOrDll:
69  large, small = win32gui.ExtractIconEx(path,nIcon)

73  hbmp = win32ui.CreateBitmap()
74  hbmp.CreateCompatibleBitmap(hdc, ico_x, ico_x)
75  hdc = hdc.CreateCompatibleDC()
76  hdc.SelectObject( hbmp )
77  #set the background in white
78  hdc.FillSolidRect( (0,0,ico_x,ico_x), 0xffffff )
79  #draw a icon in it
80  hdc.DrawIcon( (0,0), large[0] )
81  win32gui.DestroyIcon(large[0])
82  #convert picture in JPEG
83  hbmp.SaveBitmapFile( hdc, dirTmpFile)
84  im = Image.open(dirTmpFile)
85  else:
86  im = Image.open(path)
87 
88  try:
89  im.save(dst, "JPEG")
90  except IOError:
91  return None
92 
93  dst.seek(0)
94  return base64.b64encode( dst.read() )
95 
96  else:
97  return None

99 '''
100 Convert the Registry Date in Humain format date DD/MM/YYYY HH:MM:SS
101 @param date: date in registry format to convert
102 @param displayHour=True: if the registry date have no reached hour, set False to doesn't get 00:00:00
103 @return: convert date
104 '''
105 def windowsDateConversion(date, displayHour = True):
106 
107  if (len(date)==8 or len(date)==25): # check if it's good windows date system
108  finaldate = date[6:8]+"/"+date[4:6]+"/"+date[0:4]
109  if displayHour:
110  finaldate = finaldate +" "+date[8:10]+":"+date[10:12]+":"+date[12:14]
111  else: # if the date is not in registry format
112  finaldate = date
113 
114  return finaldate

116 '''
117 Calculate the size of folder
118 @param folder: the path of the folder to calculate
119 @return: the size of the folder in Ko
120 '''
121 def sizeFolder(folder):
122  size = 0
123  for (current, subFolder, files) in os.walk(folder):
124  size = size + sum( os.path.getsize( os.path.join(current, file) ) for file in files )
125  return size/1024 #Ko
