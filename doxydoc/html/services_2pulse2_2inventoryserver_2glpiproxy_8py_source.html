57 class FusionErrorHandler(_ErrorHandler):
58  """
59  Response parsing on check of occurence an error element on XML format.
60  """
61  # An example of error message :
62 
63  # <?xml version="1.0" encoding="UTF-8"?>
64  # <REPLY>
65  # <ERROR>XML not well formed!</ERROR>
66  # </REPLY>
67 
68  def _parse(self, response):
69  try:
70  dom = parseString(response)
71  for node in dom.getElementsByTagName('ERROR'):
72  if node.nodeType == node.ELEMENT_NODE:
73  self._message.append('An error occurred while talking with GLPI (details follow)')
74  self._message.append("Error was: %s" % str(node.firstChild.nodeValue))
75  except Exception, exc:
76  self._message.append('An error occurred while talking with GLPI (details follow)')
77  self._message.append('Raw error was: %s' % str(response))
78  self._message.append('With exception: %s' % str(exc))
