104  def get(self, offline=False):
105  """
106  Gets a dictionnary containing complete customer licensing info.
107 
108  @param offline: temp file check only
109  @type offline: bool
110 
111  @return: customer info
112  @rtype: Deferred
113  """
114  if not offline:
115  agent = Agent(self.clock)
116 
117  d = agent.request("GET",
118  self.url,
119  Headers(self.headers),
120  None)
121 
122  d.addCallback(self.cb_request)
123  d.addErrback(self.eb_response)
124  d.addCallback(self.cb_parse)
125  d.addCallback(self.cache_data_into_file)
126 
127  return d
128 
129  else:
130  return maybeDeferred(self.get_cached_data_from_file)

134  def cb_request(self, response):
135  """
136  Parsing the returned content of request.
137 
138  @param response: response returned from service place
139  @type response: IResponse
140 
141  @return: parsed response
142  @rtype: Deferred
143  """
