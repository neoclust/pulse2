55  def check(self):
56  """
57  @return: True if the computer can be archived, else False
58  @rtype: bool
59  """
60  # Get computer folder and boot menu
61  self.cfolder = os.path.join(self.config.imaging_api['base_folder'],
62  self.config.imaging_api['computers_folder'],
63  self.computerUUID)
64  self.bootmenu = os.path.join(self.config.imaging_api['base_folder'],
65  self.config.imaging_api['bootmenus_folder'],
66  reduceMACAddress(self.macAddress))
67  if not os.path.exists(self.cfolder):
68  self.logger.error("Looks like computer %s is not registered, so it can't be archived" % self.computerUUID)
69  return False
70  # Build valid image list
71  for image in self.imageList:
72  ifolder = os.path.join(self.config.imaging_api['base_folder'],
73  self.config.imaging_api['masters_folder'],
74  image)
75  if isPulse2Image(ifolder):
76  self.images.append(ifolder)
77  else:
78  self.logger.error("Invalid image, won't be archived: %s" % image)
79  return True

81  def prepare(self):
82  """
83  Prepare archive creation
84  """
85  try:
86  self.archivedir = self._prepareArchiveDirectory()
87  except Exception, e:
88  self.logger.error("Can't create archive directory '%s': %s"
89  % (self.archivedir, e))
90  return False
91  try:
92  self._prepareComputerDirectories()
93  except Exception, e:
94  self.logger.error("Can't prepare directories to archive: %s" % e)
95  return False
96  try:
97  self._removeComputerFromCache()
98  except Exception, e:
99  self.logger.error("Can't remove computer from UUID cache:" % e)
100  return False
101  return True

103  def _prepareArchiveDirectory(self):
104  """
105  Prepare the directory that will be used to archive the computer data
106  """
107  if self.archive:
108  archivedir = os.path.join(self.config.imaging_api['base_folder'],
109  self.config.imaging_api['archives_folder'],
110  self.computerUUID)
111  i = 1
112  while True:
113  if not os.path.exists('%s-%d' % (archivedir, i)):
114  archivedir = '%s-%d' % (archivedir, i)
115  break
116  i += 1
117  self.logger.debug('Creating archive directory: %s' % archivedir)
118  os.mkdir(archivedir)
119  self.logger.debug('Done')
120  return archivedir
121 
122  def _prepareComputerDirectories(self):
123  """
124  Mark the directories that will be archived, so that they can't no more
125  be used by the imaging server. We append '.archiving' to the file /
126  folders name.
127  """
128  self.logger.debug('Renaming directories')
129  os.rename(self.cfolder, self.cfolder + self.ARCHIVING)
130  if os.path.exists(self.bootmenu):
131  os.rename(self.bootmenu, self.bootmenu + self.ARCHIVING)
132  for ifolder in self.images:
133  os.rename(ifolder, ifolder + self.ARCHIVING)
134  self.logger.debug('Done')

144  def run(self):
145  """
146  Start the archiving process in another thread.
147 
148  @return: Return a deferred object resulting to True if all was fine
149  @rtype: Deferred
150  """
151  def _run_success(result):
152  self.logger.debug('Successfully archived data from UUID %s'
153  % self.computerUUID)
154  return True
155 
156  def _run_error(error):
157  self.logger.error('Error while archiving data from UUID %s: %s' %
158  (self.computerUUID, error))
159  return False
160 
161  d = deferToThread(self._run)
162  d.addCallbacks(_run_success, _run_error)
163  return d

165  def _run(self):
166  """
167  Perform computer data deletion or archival
168  """
169  if not self.archive:
170  # No archival, delete anything
171  self.logger.debug('Removing computer data')
172  if os.path.exists(self.bootmenu + self.ARCHIVING):
173  os.unlink(self.bootmenu + self.ARCHIVING)
174  shutil.rmtree(self.cfolder + self.ARCHIVING)
175  for ifolder in self.images:
176  shutil.rmtree(ifolder + self.ARCHIVING)
177  else:
178  # Archival, move everything to the archive directory
179  self.logger.debug('Moving computer data to archive directory')
180  # Moving boot menu if we have one
181  if os.path.exists(self.bootmenu + self.ARCHIVING):
182  dst = os.path.join(self.archivedir,
183  os.path.basename(self.bootmenu))
184  shutil.move(self.bootmenu + self.ARCHIVING, dst)
185  # Moving computer home
186  dst = os.path.join(self.archivedir, os.path.basename(self.cfolder))
187  shutil.move(self.cfolder + self.ARCHIVING, dst)
188  # Moving Pulse 2 image
189  for ifolder in self.images:
190  dst = os.path.join(self.archivedir, os.path.basename(ifolder))
191  shutil.move(ifolder + self.ARCHIVING, dst)
192  self.logger.debug('Done')
