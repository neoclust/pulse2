34 class ResultWorker(Thread):
35 
36  def __init__(self, stop, start_polling, result_queue, retry_queue, dlp_client, **kwargs):
37  Thread.__init__(self, **kwargs)
38  self.stop = stop
39  self.start_polling = start_polling
40  self.result_queue = result_queue
41  self.retry_queue = retry_queue
42  self.dlp_client = dlp_client
43  self.retry_timeout = 20
44 
45  def run(self):
46  logger.debug("%s running" % self)
47  while not self.stop.is_set():
48  try:
49  # Check if there is some result
50  # to re send first.
51  # We want to send the results in
52  # the same order as the execution
53  if not self.retry_queue.empty():
54  queue = self.retry_queue
55  # Nothing to re-send, get a new
56  # result
57  else:
58  queue = self.result_queue
59  result = queue.get(False)
60  except Queue.Empty:
61  # When results queues are empty
62  # we want to poll the DLP
63  self.start_polling.set()
64  # Nothing to do
65  # Waiting for new result...
66  self.stop.wait(10)
67  else:
68  # Error while sending the result
69  if not self.dlp_client.send_result(result):
70  self.retry_queue.put(result)
71  # Wait a little before retrying
72  self.stop.wait(self.retry_timeout)
73  queue.task_done()
74  logger.debug("Exiting from %s" % self)

78 
79  def __init__(self, stop, step_queue, result_queue, watchdog_queue, **kwargs):
80  Thread.__init__(self, **kwargs)
81  self.stop = stop
82  self.step_queue = step_queue
83  self.result_queue = result_queue
84  self.watchdog_queue = watchdog_queue
85 
86  def run(self):
87  logger.debug("%s running" % self)
88  while not self.stop.is_set():
89  try:
90  step = self.step_queue.get(False)
91  except Queue.Empty:
92  # Waiting for new step...
93  self.stop.wait(10)
94  else:
95  try:
96  step.can_run()
97  except CommandFailed:
98  # Command is failed, forget step
99  # Should not happen
100  self.watchdog_queue.put(time.time())
101  else:
102  result = step.start()
103  if result.send:
104  self.result_queue.put(result)
105  logger.debug(result)
106  if result.is_success or not step.required:
107  # If the step is not required (in non_fatal_steps)
108  # continue as if the step result is a success
109  step.command.next_step()
110  else:
111  # If step failed command is marked as failed
112  # and we wait to poll the command again
113  # from the scheduler
114  step.command.failed = True
115  logger.error("%s failed." % step.command)
116  self.watchdog_queue.put(time.time())
117  self.step_queue.task_done()
118  logger.debug("Exiting from %s" % self)

154  now = time.time()
155  logger.debug("Re-lock scheduled to %s" % datetime.fromtimestamp(self.scheduled_to).strftime("'%Y-%m-%d %H:%M:%S'"))
156  if now > self.scheduled_to and self.queues.empty():
157  logger.info("Watchdog timeout reached")
