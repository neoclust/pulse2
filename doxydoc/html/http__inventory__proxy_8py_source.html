44 def makeSSLContext(verifypeer, cacert, localcert, log = False):
45  """
46  Make the SSL context for the server, according to the parameters
47 
48  @returns: a SSL context
49  """
50  logger = logging.getLogger()
51  if verifypeer:
52  fd = open(localcert)
53  localCertificate = ssl.PrivateCertificate.loadPEM(fd.read())
54  fd.close()
55  fd = open(cacert)
56  caCertificate = ssl.Certificate.loadPEM(fd.read())
57  fd.close()
58  ctx = localCertificate.options(caCertificate)
59  ctx.verify = True
60  ctx.verifyDepth = 9
61  ctx.requireCertification = True
62  ctx.verifyOnce = True
63  ctx.enableSingleUseKeys = True
64  ctx.enableSessions = True
65  ctx.fixBrokenPeers = False
66  if log:
67  logger.debug("CA certificate informations: %s" % cacert)
68  logger.debug(caCertificate.inspect())
69  logger.debug("Inventory server certificate: %s" % localcert)
70  logger.debug(localCertificate.inspect())
71  return ctx
72  else:
73  if log:
74  logger.warning("SSL enabled, but peer verification is disabled.")
75  ctx = ssl.DefaultOpenSSLContextFactory(localcert, cacert)
76  return ctx

107  else:
108  self.uri = "http://" + self.uri

111  host = parsed[1]

113  if ':' in host:
114  host, port = host.split(':')
115  port = int(port)
116  rest = urlparse.urlunparse(('', '') + parsed[2:])
117  if not rest:
118  rest = rest + '/'
119  class_ = self.protocols[protocol]
120  headers = self.getAllHeaders().copy()
121  if 'host' not in headers:
122  headers['host'] = host
123  self.content.seek(0, 0)
124  s = self.content.read()
125 
126  logger = logging.getLogger()
127 
128  logger.debug("\nOcs Report Received");
129 
130  # We will unzip the XML File and parsing it only if needed
131  if self.config.getocsdebuglog or self.config.improve:
132  # xml file unzip
133  try:
134  decomp = decompressobj()
135  sUnpack = decomp.decompress(s)
136  if decomp.unused_data:
137  logger.warn("The zip content seems to be bad.")
138  logger.debug("The remaining bytes are : %s"%(decomp.unused_data))
139  except Exception, e:
140  logger.error("Failed during decompression.")
141  logger.error(str(e))
142  raise e
143 
144  # Get the query type
145  try:
146  query = re.search(r'<QUERY>([\w-]+)</QUERY>', sUnpack).group(1)
147  except AttributeError, e:
148  query = 'FAILS'
149

151  if query == 'INVENTORY' :
152  try:
153  if sys.platform[0:3] == "win": #It's here because I need Pulse2InventoryProxyConfig be initialited before improve packtage be initialat
154  from improveWin import improveXML,getOcsDebugLog

156  from improveMac import improveXML,getOcsDebugLog # pyflakes.ignore
157  else:

161  # Add Ocs Log File

164  if self.config.improve:

167  logger.debug("\t\tInformations add")
168  except ImportError:

170

178  logger.debug("\t\tcompressed length: " + str(len(s)))

181  logger.error(str(e))
182  raise e
183
