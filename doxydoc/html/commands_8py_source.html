187  def getCohIds(self, target_uuids=[]):
188  """
189  Returns the list of commands_on_host linked to this command
190  If list of target_uuids, returns only uuids of this list
191  """
192  session = sqlalchemy.orm.create_session()
193  myCommandOnHosts = session.query(CommandsOnHost)
194  if target_uuids:
195  myCommandOnHosts = myCommandOnHosts.join(Target)
196  myCommandOnHosts = myCommandOnHosts.filter(Target.target_uuid.in_(target_uuids))
197  myCommandOnHosts = myCommandOnHosts.filter(CommandsOnHost.fk_commands ==
198  self.getId())
199  session.close()
200  return myCommandOnHosts.all()

297  groups = CoHManager.setCoHsStateStopped(cohs)
298  session = sqlalchemy.orm.create_session()
299 
300  for cmd_id, count in groups.items():

309 def stopCommand(c_id):
310  """
311  Stop a command, by stopping all its related commands_on_host.
312  @returns: the list of all related commands_on_host
313  @rtype: list
314  """
315  session = sqlalchemy.create_session()
316  myCommand = session.query(Commands).get(c_id)
317  ret = []
318  for cmd in myCommand.getCohIds():
319  ret.append(cmd)
320  stopCommandOnHost(cmd.id)
321  session.close()
322  return ret
