105  def set_daemon_pid(self, args):
106  """
107  Checks the process and creates its PID file.
108 
109  @param args: command
110  @type args: list
111 
112  @return: True if process found
113  @rtype: bool
114  """
115  # A little hack - because autossh in ps list has another path
116  for i, a in enumerate(args):
117  if a=="/usr/bin/autossh":
118  args[i] = "/usr/lib/autossh/autossh"
119  break
120 
121  for p in psutil.process_iter():
122  if p.cmdline == args :
123  self.daemon_proc = p
124  self.logger.info("SSH Tunnel: pid=%d" % p.pid)
125  self.write_pid_file()
126  return True
127  self.logger.warn("SSH Tunnel: Can't find the pid for: %s" % " ".join(args))
128  return False

167  """
168  Gets the PID of process from PID file.
169 
170  @return: PID or None if not exists
171  @rtype: int
172  """
173

207 class TunnelBuilder(object):
208 
209  """
210  Provides the forking of given process.
211 
212  Spawns a process and controls its life-time.
213  Instance of this class must be referenced as singleton
214  to avoid the duplicities of controlled process.
215  """
216 
217  pid_control = None
218  port = 0
219  config = None
220 
221  def __init__(self, config):
222  """
223  Initialize the builder.
224 
225  The builder checks immediatelly a occurence of PID file and process;
226  When a process is already established, creates a expirator
227  which provides a responding life-time.
228  If the difference between start time of checked process
229  and current time is greater than timeout, this process
230  will be considered as already expired.
231 
232  @param config: support config instance
233  @type config: PluginConfig
234  """
235  self.config = config
236  self.pid_control = PIDControl(config.pid_path)
237  self.logger = logging.getLogger()
238 
239  if self.established:
240  pid_date = self.pid_control.get_pid_date()
241  delay = pid_date + self.config.session_timeout - time.time()
242  if delay > 0:
243  self.logger.info("SSH Tunnel: Already established - session will be expired at: %d seconds" % int(delay))
244  else:
245  delay = 1
246  # !!! do not put delay = 0 !!!
247  self.logger.info("SSH Tunnel: Found tunnel expired")
248 
249  self._do_expire(delay)
