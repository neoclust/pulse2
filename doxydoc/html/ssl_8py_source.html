36 def makeSSLContext(verifypeer, cacert, localcert, cb, log = True):
37  """
38  Make the SSL context for the server, according to the parameters
39 
40  @returns: a SSL context
41  """
42  logger = logging.getLogger()
43  if verifypeer:
44  # I use twisted certificate loading function, else our CA and our
45  # private key don't match, I didn't find out why
46  fd = open(localcert)
47  localCertificate = ssl.PrivateCertificate.loadPEM(fd.read())
48  fd.close()
49  fd = open(cacert)
50  caCertificate = ssl.Certificate.loadPEM(fd.read())
51  fd.close()
52  ctx = localCertificate.options(caCertificate)
53  ctx.verify = True
54  ctx.verifyDepth = 9
55  ctx.requireCertification = True
56  ctx.verifyOnce = True
57  ctx.enableSingleUseKeys = True
58  ctx.enableSessions = True
59  ctx.fixBrokenPeers = False
60  if log:
61  logger.debug("CA certificate informations: %s" % cacert)
62  logger.debug(caCertificate.inspect())
63  logger.debug("Inventory server certificate: %s" % localcert)
64  logger.debug(localCertificate.inspect())
65  return ctx.getContext()
66  else:
67  if log:
68  logger.warning("SSL enabled, but peer verification is disabled.")
69  ctx = SSL.Context(SSL.SSLv23_METHOD)
70  ctx.use_privatekey_file(localcert)
71  ctx.use_certificate_file(cacert)
72  return ctx
