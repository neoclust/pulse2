153  protected function drawDisc($cy, $colorArray, $mode) {
154  // Get graphical obects
155  $img = $this->plot->getImg();
156 
157  $i = 0;
158  $oldAngle = 0;
159  $percentTotal = 0;
160 
161  foreach ($this->percent as $a) {
162  list ($percent, $point) = $a;
163 
164  // If value is null, don't draw this arc
165  if ($percent <= 0) {
166  continue;
167  }
168 
169  $color = $colorArray[$i % count($colorArray)];
170 
171  $percentTotal += $percent;

173

197 
198  foreach ($this->percent as $a) {
199  list ($percent, $point) = $a;
200 
201  // If value is null, the arc isn't drawn, no need to display percent
202  if ($percent <= 0) {
203  continue;
204  }
205 
206  $percentTotal += $percent;
207  $angle2 = $percentTotal * 2 * M_PI / 100;
208 
209  $angle = $angle1 + ($angle2 - $angle1) / 2;
210  $label = number_format($percent) . "%";
211 
212  $x = cos($angle) * ($this->pieWidth + 35) / 2 + $this->pieCenterX;
213  $y = sin($angle) * ($this->pieHeight + 35) / 2 + $this->pieCenterY;
214 
215  $text->printText($img, $x, $y, $this->plot->getTextColor(), $label, $text->fontCondensed, $text->HORIZONTAL_CENTER_ALIGN | $text->VERTICAL_CENTER_ALIGN);
216 
217  $angle1 = $angle2;
