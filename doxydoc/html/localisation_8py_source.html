88  def distHaversine(self, p1, p2):
89  """
90  # Calcul de la distance (en km) entre 2 points specifies par leurs
91  # latitude/longitude avec la formule de Haversine
92  #
93  # de : Haversine formula - R. W. Sinnott, "Virtues of the Haversine",
94  # Sky and Telescope, vol 68, no 2, 1984
95  # http://www.census.gov/cgi-bin/geo/gisfaq?Q5.1
96  #
97  """
98  dLat = p2.lat - p1.lat
99  dLong = p2.lon - p1.lon
100  a = sin(dLat/2) * sin(dLat/2) + cos(p1.lat) * cos(p2.lat) * sin(dLong/2) * sin(dLong/2)
101  c = 2 * atan2(sqrt(a), sqrt(1-a))
102  d = rt * c
103  return d
