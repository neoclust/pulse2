308  def delMachine(self, uuid):
309  """
310  delete a machine from dyngroup database
311  and clears all static groups and imaging group linked
312  to the machine
313  """
314  logging.getLogger().info('Removing all groups associated to machine %s.' % uuid)
315  session = create_session()
316  # First get machine id
317  mid = session.query(Machines.id).filter_by(uuid = uuid).scalar()
318  if not mid:
319  logging.getLogger().info('Machine not found in dyngroup database, skipping.')
320  return False
321 
322  # Deleting all entries in Results and ProfileResults
323  try:
324  session.query(Results).filter_by(FK_machines=mid).delete()
325  session.query(ProfilesResults).filter_by(FK_machines=mid).delete()
326  session.query(Machines).filter_by(id = mid).delete()
327  except Exception, e:
328  logging.getLogger().error('Cannot delete machine %s from associated groups.' % uuid)
329  logging.getLogger().error(str(e))
330  return False
331  session.close()
332 
333  return True

630  def __merge_join_query(self, select_from, join_tables):
631  """
632  utility function to merge a list of tables (join_tables) to a previous select_from
633  and return the select_from
634  """
635  for table in join_tables:
636  if type(table) == list:
637  if table[1]: # outerjoin
638  if len(table) > 2: # specific on clause
639  select_from = select_from.outerjoin(table[0], table[2])
640  else:
641  select_from = select_from.outerjoin(table[0])
642  else: # normal join
643  if len(table) > 2: # specific on clause
644  select_from = select_from.join(table[0], table[2])
645  else: # hum... why did u use a list!
646  select_from = select_from.join(table[0])
647  else:
648  select_from = select_from.join(table)
649  return select_from

660  def __insert_into_machines_and_results(self, connection, computers, groupid, type = 0):
661  """
662  This function is called by reload_group and addmembers_to_group to
663  update the Results and Machines tables of the database.
664 
665  @param computers: list of dicts with {'uuid':uuid, 'hostname':name}
666  @type computers: list
667  """
668  session = create_session()
669 
670  # transform to dictionnary on format {"uuid": "hostname",}
671  comp_dict = dict([(c["uuid"], c["hostname"]) for c in computers])
672  # existing machines
673  existing = session.query(Machines).filter(
674  self.machines.c.uuid.in_(comp_dict.keys())
675  ).all()
676  results_ids = []
677 
678  for comp in existing :
679  # if the hostname was changed, we change this name in the db
