300 def SupEsp (obj):
301  """
302  Call the function SupEspLi if the object is a list, SupEspDi if it's a dict or if it's a string return the string without the space.
303  """
304  if type(obj) == type([]):
305  SupEspLi(obj)
306  elif type(obj) == type({}):
307  SupEspDi(obj)
308  elif type(obj) == type(""):
309  lobj=[obj]
310  lobj[0]=lobj[0].strip()
311  obj=lobj[0]
312  return obj
