125  $barColorSet = $palette->barColorSet;
126  $barColorSet->reset();
127 
128  $serieCount = count($serieList);
129  for ($j = 0; $j < $serieCount; $j++) {

131  $pointList = $serie->getPointList();
132  $pointCount = count($pointList);
133  reset($pointList);
134 
135  // Select the next color for the next serie
136  if (!$this->config->getUseMultipleColor()) {
137  $color = $barColorSet->currentColor();
138  $shadowColor = $barColorSet->currentShadowColor();
139  $barColorSet->next();
140  }
141 
142  $rowHeight = ($graphArea->y2 - $graphArea->y1) / $pointCount;
143  for ($i = 0; $i < $pointCount; $i++) {
144  $y = $graphArea->y2 - $i * $rowHeight;
145 
146  $point = current($pointList);
147  next($pointList);
148 
149  $value = $point->getY();
150

152 
153  // Bar dimensions
154  $yWithMargin = $y - $rowHeight * $this->emptyToFullRatio;

156  $barWidth = $rowWidthWithMargin / $serieCount;
157  $barOffset = $barWidth * $j;

159  $y2 = $yWithMargin - $barOffset - 1;
160

166  }
167 
168  // Draw caption text on bar
169  if ($this->config->getShowPointCaption()) {
170  $text->printText($img, $xmax + 5, $y2 - $barWidth / 2, $this->plot->getTextColor(), $value, $text->fontCondensed, $text->VERTICAL_CENTER_ALIGN);
171  }
172 
173  // Draw the horizontal bar
174  imagefilledrectangle($img, $graphArea->x1 + 1, $y1, $xmax, $y2, $shadowColor->getColor($img));
