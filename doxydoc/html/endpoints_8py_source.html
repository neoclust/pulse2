323  def process_inventory(self, inventory, from_ip):
324  """
325  Initial stage when an inventory of machine received.
326 
327  At first, we check if machine already exists in inventory.
328  In this case, its UUID is immediatelly returned; if not,
329  received inventory collection is transformed to fusion/OCS
330  format and sent to inventory server.
331  Because inscription of a new machine takes always few seconds,
332  a lag defined in config allows to wait until our new machine
333  is saved into the inventory database.
334  After this delay, a new request to get UUID of this machine
335  is repeated.
336 
337  @param inventory: inventory collection
338  @type inventory: list
339 
340  @param from_ip: IP address of connected machine
341  @param from_ip: str
342 
343  @return: UUID of machine or False if any fail occured
344  @rtype: Deferred
345  """
346 
347  hostname, system, osversion, networks = inventory
348  macs = [mac for (name, ip, mac, mask) in networks]
349  netmask = self.get_netmask(networks, from_ip)
350 
351  d = self._get_machine_uuid(hostname, macs)
352  # UUID of machine
353  d.addCallback(self.get_uuid_callback, hostname, macs, inventory)
354  # IP address update for UUID (msc.target table)
355  d.addCallback(self.update_target_ip, from_ip, netmask)
356  @d.addErrback
357  def eb_uuid(failure):
358  self.logger.warn("Getting of UUID for machine '%s' failed: %s" % (hostname, str(failure)))
359  return failure
360 
361  # Light Pull stage
362  d.addCallback(self.check_pending_commands, hostname)
363  d.addCallback(self.launch_pending_commands, hostname)
364  d.addCallback(self.catch_results, hostname)
365  @d.addErrback
366  def eb_light_pull(failure):
367  self.logger.warn("Pending commands launching for machine '%s' failed: %s" % (hostname, str(failure)))
368  return failure
369 
370  return d
