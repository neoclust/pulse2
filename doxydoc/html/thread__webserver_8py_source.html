52 def initialize(config):
53  """
54  Init package server API, and start listening to the network.
55 
56  @rtype: bool
57  @return: return False if initialization failed
58  """
59  ret = True
60  logger = logging.getLogger()
61 
62  port = int(config.port)
63 
64  server = MyServer()
65  services = []
66  if len(config.mirrors) > 0:
67  for mirror_params in config.mirrors:
68  # If source dosesnt exist, create it
69  if not os.path.isdir(mirror_params['src']):
70  os.mkdir(mirror_params['src'])
71  m_api = Mirror(mirror_params['mount_point'], mirror_params['mount_point'])
72  server.register(m_api, mirror_params['mount_point'])
73  services.append({'type':'mirror', 'mp':mirror_params['mount_point'], 'server':config.public_ip, 'port':config.port, 'proto':config.proto, 'src':mirror_params['src'], 'mirror_mp': '%s_files'%(mirror_params['mount_point']), 'mirror_url':'%s://%s:%s%s_files'%(config.proto, config.public_ip, config.port, mirror_params['mount_point']), 'url':'%s://%s:%s%s'%(config.proto, config.public_ip, config.port, mirror_params['mount_point'])})
74  server.register(static.File(mirror_params['src']), mirror_params['mount_point']+'_files')
75  services.append({'type':'mirror_files', 'mp':mirror_params['mount_point']+'_files', 'server':config.public_ip, 'port':config.port, 'proto':config.proto, 'src':mirror_params['src']})
76 
77  if len(config.package_api_get) > 0:
78  for mirror_params in config.package_api_get:
79  p_api = PackageApiGet(mirror_params['mount_point'], mirror_params['mount_point'])
80  server.register(p_api, mirror_params['mount_point'])
81  services.append({'type':'package_api_get', 'mp':mirror_params['mount_point'], 'server':config.public_ip, 'port':config.port, 'proto':config.proto, 'src':mirror_params['src'], 'url':'%s://%s:%s%s'%(config.proto, config.public_ip, config.port, mirror_params['mount_point'])})
82 
83  if len(config.package_api_put) > 0:
84  for mirror_params in config.package_api_put:
85  p_api = PackageApiPut(mirror_params['mount_point'], mirror_params['mount_point'], mirror_params['tmp_input_dir'])
86  server.register(p_api, mirror_params['mount_point'])
87  services.append({'type':'package_api_put', 'mp':mirror_params['mount_point'], 'server':config.public_ip, 'port':config.port, 'proto':config.proto, 'src':mirror_params['src'], 'url':'%s://%s:%s%s'%(config.proto, config.public_ip, config.port, mirror_params['mount_point'])})
88 
89  if 'mount_point' in config.user_package_api:
90  mirror = UserPackageApi(services, config.user_package_api['mount_point'], config.up_assign_algo)
91  server.register(mirror, config.user_package_api['mount_point'])
92  services.append({'type':'user_package_api', 'mp':config.user_package_api['mount_point'], 'server':config.public_ip, 'port':config.port, 'proto':config.proto})
93 
94  if 'mount_point' in config.mirror_api:
95  mirror = MirrorApi(services, config.mirror_api['mount_point'], config.mm_assign_algo)
96  server.register(mirror, config.mirror_api['mount_point'])
97  services.append({'type':'mirror_api', 'mp':config.mirror_api['mount_point'], 'server':config.public_ip, 'port':config.port, 'proto':config.proto})
98  else:
99  logger.warn('package server initialized without mirror api')
100 
101  if 'mount_point' in config.scheduler_api:
102  scheduler = SchedulerApi(config.scheduler_api['mount_point'], config.scheduler_api)
103  server.register(scheduler, config.scheduler_api['mount_point'])
104  services.append({'type':'scheduler_api', 'mp':config.scheduler_api['mount_point'], 'server':config.public_ip, 'port':config.port, 'proto':config.proto})
105  logger.info("package server initialized with scheduler api")
106 
107  if config.imaging_api:
108  try:
109  imaging = ImagingApi(config.imaging_api['mount_point'], config)
110  server.register(imaging, config.imaging_api['mount_point'])
111  services.append({
112  'type' : 'imaging',
113  'mp' : config.imaging_api['mount_point'],
114  'server' : config.public_ip,
115  'port' : config.port,
116  'proto' : config.proto
117  })
118  logger.info("Package Server initialized with imaging API")
119  try:
120  PXEProxy(config, imaging.api)
121  except CannotListenError :
122  logger.error("PXE proxy: start failed")
123  logger.error("PXE Proxy: Adress already in use: old LRS imaging-server still installed and not yet stopped")
124  logger.error("PXE proxy: Please verify your configuration and restart the service")
125 
126  except Exception, e:
127  logger.exception("Imaging error: %s" % e)
128  logger.error("PXE imaging service initialization failed, exiting.")
129  else :
130  logger.info("Package Server initialized with PXE imaging API")
131 
132  except Exception, e:
133  logger.exception("Imaging error: %s" % e)
134  logger.error("Error while initializing the imaging API")
135  logger.error("Package Server will run WITHOUT the imaging API")
136 
137  desc = Description(services)
138  server.register(desc, '/desc')
139  Common().setDesc(services)
140 
141  try:
142  if config.enablessl:
143  pulse2.xmlrpc.OpenSSLContext().setup(config.localcert, config.cacert, config.verifypeer)
144  twisted.internet.reactor.listenSSL(
145  port,
146  P2PSite(server),
147  interface = config.bind,
148  contextFactory = pulse2.xmlrpc.OpenSSLContext().getContext()
149  )
150  logger.info('activating SSL mode')
151  else:
152  twisted.internet.reactor.listenTCP(
153  port,
154  twisted.web.server.Site(server),
155  interface = config.bind
156  )
157  logger.info('package server listening on %s:%d' % (config.bind, port))
158  except Exception, e:
159  logger.error('can\'t bind to %s:%d' % (config.bind, port))
160  logger.error(e)
161  ret = False
162 
163  return ret
