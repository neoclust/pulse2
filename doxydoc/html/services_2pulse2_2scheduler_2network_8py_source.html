41  @classmethod
42  def prepare_target(cls, target):
43  """
44  Convert the scheduler target format to common format.
45 
46  @param target: target info
47  @type target: dict
48 
49  @return: target common format
50  @rtype: tuple
51  """
52  hostname = u_decode(target["shortname"])
53  fqdn = u_decode(target["fqdn"])
54  ips = u_list_decode(target["ips"])
55  macs = u_list_decode(target["macs"])
56  if "netmasks" in target :
57  netmasks = u_list_decode(target["netmasks"])
58 
59  i = 0
60  ifaces = []
61  for ip in ips :
62  iface = {}
63  iface["ip"] = ip
64  if i < len(macs) :
65  iface["mac"] = macs[i]
66  if i < len(netmasks) :
67  iface["netmask"] = netmasks[i]
68  ifaces.append(iface)
69  i += 1
70 
71  return (hostname, fqdn, ifaces)
